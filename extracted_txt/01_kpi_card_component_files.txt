const STATUS_COLOR = { RUN: "#22d3a0", IDLE: "#fbbf24", DOWN: "#f87171" };
const STATUS_BG    = { RUN: "#022c22", IDLE: "#2c1d00", DOWN: "#2d0000" };

/** defects 배열에서 class_name으로 색상 조회 */
function defectColor(clsName, defects) {
  return defects.find(d => d.class_name === clsName)?.color ?? "#6b7fa3";
}

// ── 유틸: HTML 이스케이프 (XSS 방지) ────────────────────
function esc(str) {
  return String(str)
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;");
}

// ── 유틸: 숫자 포맷 ─────────────────────────────────────
const fmt  = (n) => Number(n).toLocaleString("ko-KR");
const pct  = (n, dec = 1) => Number(n).toFixed(dec) + "%";

// ════════════════════════════════════════════════════════
// 🖼️ HTML 렌더러
// ════════════════════════════════════════════════════════

/** KPI 카드 HTML 생성 (재사용 가능) */
function kpiCardHTML({ label, value, sub = "", accent }) {
  return `
    <div class="kpi-card" style="border:1px solid ${accent}33; box-shadow:0 0 20px ${accent}14;">
      <div class="kpi-label">${esc(label)}</div>
      <div class="kpi-value" style="color:${accent}">${esc(value)}</div>
      ${sub ? `<div class="kpi-sub">${esc(sub)}</div>` : ""}
    </div>`;
}

// ── SCREEN 1: 작업자 ─────────────────────────────────────
function renderWorker({ defects, ngList, summary }) {
  const maxCount = Math.max(...defects.map(d => d.count));

  // KPI 카드 데이터
  const kpis = [
    { label: "목표 생산량",  value: "12,000 pcs",      accent: "#4a9eff", sub: "오늘 LINE B" },
    { label: "현재 생산량",  value: "7,840 pcs",        accent: "#22d3a0", sub: "달성률 65.3%" },
    { label: "현재 불량률",  value: "2.8%",             accent: "#fbbf24", sub: "예측 3.4%" },
    { label: "최근 10분 NG", value: "26건",             accent: "#f87171", sub: "주요: short" },
  ];

  // Defect 바 리스트 HTML
  const defectBars = defects.map(d => `
    <div class="defect-bar-row">
      <div class="defect-bar-label">${esc(d.class_name)}</div>
      <div class="defect-bar-track">
        <div class="defect-bar-fill" style="width:${(d.count / maxCount * 100).toFixed(1)}%; background:${d.color};"></div>
      </div>
      <div class="defect-bar-count" style="color:${d.color}">${d.count}</div>
    </div>`).join("");

  // NG 로그 HTML (최대 7개)
  const ngRows = ngList.slice(0, 7).map(r => {
    const clsColor = defectColor(r.cls, defects);
    const lineId   = r.board_id.split("-")[0];
    const confPct  = Math.floor(r.conf * 100);
    return `
      <div class="ng-row">
        <span class="ng-time">${esc(r.time)}</span>
        <span class="ng-line">LINE ${esc(lineId)}</span>
        <span class="ng-badge" style="background:${clsColor}22; color:${clsColor}; border:1px solid ${clsColor}44">${esc(r.cls)}</span>
        <span class="ng-conf">${confPct}%</span>
        <span class="ng-tag">NG</span>
      </div>`;
  }).join("");

  return `
    <div class="screen">
      <!-- 경고 배너 -->
      <div class="alert-banner">
        <span class="alert-icon">⚠</span>
        <div style="flex:1">
          <div class="alert-title">LINE B — short 불량 급증 감지</div>
          <div class="alert-msg">최근 10분 26건 발생 · 즉시 점검 요망</div>
        </div>
        <span class="alert-time">14:59:21</span>
      </div>

      <!-- KPI 카드 2열(모바일)/4열(태블릿+) -->
      <div class="grid grid-4" style="gap:10px">
        ${kpis.map(kpiCardHTML).join("")}
      </div>

      <!-- 하단 2단 -->
      <div class="grid grid-chart">
        <!-- Defect 바 차트 -->
        <div class="card">
          <div class="section-title">Defect Class 실시간</div>
          ${defectBars}
        </div>

        <!-- NG 로그 -->
        <div class="card">
          <div class="section-title">실시간 NG 로그</div>
          ${ngRows}
        </div>
      </div>
    </div>`;
}

  const kpis = [
    { label: "계획 생산량", value: fmt(total_plan) + " pcs",      accent: "#60a5fa", sub: "오늘 전체" },

          <div class="line-badge" style="background:${bg}; border:1px solid ${clr}55; color:${clr}">L${esc(l.id)}</div>

            <div style="display:flex; justify-content:space-between; align-items:center; margin-bottom:5px">
              <span class="line-produced">${fmt(l.produced)} <small>pcs</small></span>
              <span class="line-status-badge" style="color:${clr}; background:${bg}; border:1px solid ${clr}55">${esc(l.status)}</span>
            </div>
            <div style="display:flex; align-items:center; gap:8px">
              <div class="progress-track">
                <div class="progress-fill" style="width:${barPct}%; background:${barColor}"></div>
              </div>

        <div class="risk-label">${esc(d.class_name)}</div>
        <div class="risk-donut" style="width:56px;height:56px; background:conic-gradient(${d.color} ${deg}deg, #1e2d40 0)">
          <div class="risk-inner" style="width:42px;height:42px">
            <span class="risk-pct" style="color:${d.color}">${(d.risk * 100).toFixed(0)}%</span>
          </div>
        </div>
        <div class="risk-count">${d.count}건</div>
      </div>`;
  }).join("");

  return `
    <div class="screen">
      <!-- ① KPI 5열 -->
      <div class="grid grid-5" style="gap:10px">
        ${kpis.map(kpiCardHTML).join("")}

        <div class="card">
          <div class="section-title">시간대별 생산량 / 불량률</div>
          <div class="chart-wrap" style="height:230px">
            <canvas id="managerHourlyChart"></canvas>
          </div>
        </div>
        <div class="card">
          <div class="section-title">라인별 현황</div>

        </div>
      </div>

      <!-- ③ Defect 리스크 게이지 -->

/**
 * 홍보 스크린의 실시간 시계 + TOP Defect 롤링 시작
 * @param {Array} defects
 */
function startPromoTimers(defects) {
  // 기존 타이머 정리
  promoTimers.forEach(clearInterval);
  promoTimers = [];

  // 발생 건수 내림차순 정렬
  const sorted = [...defects].sort((a, b) => b.count - a.count).slice(0, 3);
  let idx = 0;

  /** TOP Defect 패널 업데이트 */
  function updateTopDefect() {
    const d     = sorted[idx];
    const panel = document.getElementById("topDefectPanel");
    const rank  = document.getElementById("topDefectRank");
    const name  = document.getElementById("topDefectName");
    const count = document.getElementById("topDefectCount");
    const dots  = document.getElementById("topDefectDots");
    if (!panel) return;

    panel.style.borderColor  = d.color + "44";
    panel.style.boxShadow    = `0 0 40px ${d.color}18`;
    rank.style.background    = d.color + "18";
    rank.style.border        = `2px solid ${d.color}66`;
    rank.textContent         = `#${idx + 1}`;
    name.style.color         = d.color;
    name.textContent         = d.class_name.toUpperCase();
    count.textContent        = `${d.count}건 발생`;

    // 페이지 인디케이터 점 렌더링
    dots.innerHTML = sorted.map((s, i) =>
      `<div class="dot ${i === idx ? "active" : ""}" style="--dot-color:${s.color}"></div>`
    ).join("");

    idx = (idx + 1) % sorted.length;
  }

  updateTopDefect(); // 즉시 1회 실행
  promoTimers.push(setInterval(updateTopDefect, 3000)); // 3초마다 전환

  /** 실시간 시계 업데이트 */
  function updateClock() {
    const el = document.getElementById("promoClock");
    if (el) el.textContent = new Date().toLocaleTimeString("ko-KR");
  }

  updateClock();
  promoTimers.push(setInterval(updateClock, 1000)); // 1초마다 갱신
}

/**
 * 탭을 활성화하고 해당 화면을 렌더링합니다.
 * @param {string} tabId - "worker" | "qa" | "manager" | "promo"
 */
function activateTab(tabId) {
  // 홍보 스크린 타이머 정리 (다른 탭으로 이동 시)
  if (currentTab === "promo") {
    promoTimers.forEach(clearInterval);
    promoTimers = [];
  }

  currentTab = tabId;

  // 탭 버튼 활성 스타일 적용
  document.querySelectorAll(".tab-btn").forEach(btn => {
    const isActive = btn.dataset.tab === tabId;
    btn.classList.toggle("active", isActive);
    if (isActive) {
      const tabDef = TABS.find(t => t.id === tabId);
      btn.style.setProperty("--tab-color", tabDef?.color ?? "#22d3a0");
    }
  });

  // 캐시 데이터로 화면 렌더링
  const { lines, defects, hourly, ngList, summary } = cache;
  const content = document.getElementById("content");

  // 화면별 HTML 렌더링
  switch (tabId) {
    case "worker":  content.innerHTML = renderWorker({ defects, ngList, summary });   break;
    case "qa":      content.innerHTML = renderQA({ defects, hourly, ngList });        break;
    case "manager": content.innerHTML = renderManager({ lines, defects, hourly, summary }); break;
    case "promo":   content.innerHTML = renderPromo({ defects, hourly });             break;
  }

  // 화면별 차트 초기화 (DOM 삽입 후 실행됨)
  requestAnimationFrame(() => {
    switch (tabId) {
      case "worker":
        // 작업자 화면: 차트 없음 (바는 CSS로 렌더링)
        break;
      case "qa":
        renderDefectRateChart("qaLineChart", hourly);
        renderDefectPieChart("qaPieChart", defects);
        break;
      case "manager":
        renderHourlyChart("managerHourlyChart", hourly);
        break;
      case "promo":
        renderPromoLineChart("promoLineChart", hourly);
        startPromoTimers(defects);
        break;
    }
  });
}

/**
 * 앱 초기화 — DOM 로드 후 실행
 * 1. API에서 전체 데이터 로드
 * 2. 탭 이벤트 리스너 등록
 * 3. 기본 탭(관리자) 활성화
 */
async function initApp() {
  try {
    // 모든 API 병렬 호출 (fetchAll: api.js)
    cache = await fetchAll();

    // 탭 버튼 클릭 이벤트 등록
    document.querySelectorAll(".tab-btn").forEach(btn => {
      btn.addEventListener("click", () => activateTab(btn.dataset.tab));
    });

    // 기본 탭 활성화
    activateTab("manager");

  } catch (err) {
    // API 오류 시 에러 메시지 표시
    document.getElementById("content").innerHTML = `
      <div style="display:flex; flex-direction:column; align-items:center; justify-content:center; height:60vh; color:#f87171; gap:12px">
        <div style="font-size:32px">⚠</div>
        <div>API 연결 실패: ${esc(err.message)}</div>
        <div style="font-size:12px; color:#4a5d7a">FastAPI 서버가 실행 중인지 확인하세요</div>
        <code style="font-size:11px; color:#6b7fa3">uvicorn backend.main:app --reload</code>
      </div>`;
  }
}

// DOM 준비 완료 후 앱 초기화 실행
