    99	/**
   100	 * 내부 상태값(value) → tone(normal/warning/danger/neutral) 반환
   101	 * @param {string|number} value  - 상태값 또는 수치
   102	 * @param {object} [opts]
   103	 *   opts.thresholds  - [{max, tone}] 오름차순. 미지정 시 기본 규칙 사용
   104	 *   opts.invert      - true: 낮을수록 danger (자재 잔량 등)
   105	 */
   106	function resolveTone(value, opts = {}) {
   107	  // 문자열 상태값 매핑
   108	  const STATUS_TONE = {
   109	    RUN:    "normal",
   110	    IDLE:   "warning",
   111	    DOWN:   "danger",
   112	    normal: "normal",
   113	    warning:"warning",
   114	    danger: "danger",
   115	  };
   116	  if (typeof value === "string") return STATUS_TONE[value] ?? "neutral";
   117	
   118	  // 수치 기반 임계치
   119	  const num = Number(value);
   120	  const thresholds = opts.thresholds ?? [
   121	    { max: 39,  tone: opts.invert ? "normal"  : "danger"  },
   122	    { max: 69,  tone: "warning" },
   123	    { max: 100, tone: opts.invert ? "danger"  : "normal"  },
   124	  ];
   125	  for (const t of thresholds) {
   126	    if (num <= t.max) return t.tone;
   127	  }
   128	  return opts.invert ? "danger" : "normal";
   129	}

   219	function compareDetailValues(a, b, type = "text", direction = "asc") {
   220	  const dir = direction === "desc" ? -1 : 1;
   221	
   222	  if (type === "datetime") {
   223	    const aTime = new Date(a).getTime();
   224	    const bTime = new Date(b).getTime();
   225	    const safeA = Number.isFinite(aTime) ? aTime : -Infinity;
   226	    const safeB = Number.isFinite(bTime) ? bTime : -Infinity;
   227	    return (safeA - safeB) * dir;
   228	  }
   229	
   230	  if (type === "number" || type === "percent") {
   231	    const safeA = Number.isFinite(Number(a)) ? Number(a) : -Infinity;
   232	    const safeB = Number.isFinite(Number(b)) ? Number(b) : -Infinity;
   233	    return (safeA - safeB) * dir;
   234	  }
   235	
   236	  return String(a ?? "").localeCompare(String(b ?? ""), "ko-KR") * dir;
   237	}

   319	 * 프로그레스 바 블록 HTML 반환 (재사용 가능)
   320	 * @param {object} p
   321	 *   p.label         - 항목 라벨
   322	 *   p.value         - 현재 값 (숫자)
   323	 *   p.max           - 최대 값 (기본 100)
   324	 *   p.unit          - 단위 (기본 "%")
   325	 *   p.helperText    - 보조 텍스트
   326	 *   p.tone          - 직접 tone 지정 시 임계치 계산 생략
   327	 *   p.invert        - true: 낮을수록 danger
   328	 *   p.thresholds    - 사용자 정의 임계치 배열
   329	 */
   330	function progressBarHTML(p) {
   331	  const max    = p.max  ?? 100;
   332	  const unit   = p.unit ?? "%";
   333	  const pct    = Math.min(Math.max((p.value / max) * 100, 0), 100).toFixed(1);
   334	  const tone   = p.tone ?? resolveTone(p.value, {
   335	    invert:     p.invert,
   336	    thresholds: p.thresholds,
   337	  });
   338	  const displayVal = typeof p.value === "number"
   339	    ? (Number.isInteger(p.value) ? p.value : p.value.toFixed(1))
   340	    : p.value;
   341	  const helperHTML = p.helperText
   342	    ? `<div class="progress-block-helper">${esc(p.helperText)}</div>`
   343	    : "";
   344	
   345	  // 색상 강조: 값 색상도 tone 연동
   346	  const valColor = KPI_STATUS_COLOR[tone] ?? KPI_STATUS_COLOR.neutral;
   347	
   348	  return `
   349	    <div class="progress-block">
   350	      <div class="progress-block-header">
   351	        <span class="progress-block-label">${esc(p.label)}</span>
   352	        <span class="progress-block-value" style="color:${valColor}">${esc(displayVal)}${esc(unit)}</span>
   353	      </div>
   354	      <div class="progress-block-track" role="progressbar"
   355	           aria-valuenow="${p.value}" aria-valuemin="0" aria-valuemax="${max}"
   356	           aria-label="${esc(p.label)} ${esc(displayVal)}${esc(unit)}">
   357	        <div class="progress-block-fill progress-block-fill--${tone}" style="width:${pct}%"></div>
   358	      </div>
   359	      ${helperHTML}
   360	    </div>`;
   361	}

   574	 *   opts.cutout         - 도넛 두께 (기본 "62%")
   575	 */
   576	function renderDonutChart(canvasId, items, opts = {}) {
   577	  const cutout = opts.cutout ?? "62%";
   578	  const { chartItems, total } = prepareDonutDisplayData(items, opts);
   579	
   580	  // 중심 수치 인라인 플러그인 (opts.centerText 있을 때만 등록)
   581	  const plugins = [];
   582	  if (opts.centerText != null) {
   583	    plugins.push({
   584	      id: "donutCenter",
   585	      beforeDraw(chart) {
   586	        const { ctx: c, chartArea } = chart;
   587	        const cx = chartArea.left + chartArea.width  / 2;
   588	        const cy = chartArea.top  + chartArea.height / 2;
   589	        const hasSub = Boolean(opts.centerSubText);
   590	        c.save();
   591	        c.textAlign    = "center";
   592	        c.textBaseline = "middle";
   593	        c.fillStyle    = CHART_TOKENS.textMain;
   594	        c.font         = `700 16px ${CHART_TOKENS.fontMono}`;
   595	        c.fillText(String(opts.centerText), cx, cy - (hasSub ? 9 : 0));
   596	        if (hasSub) {
   597	          c.fillStyle = CHART_TOKENS.textLabel;
   598	          c.font      = `400 10px ${CHART_TOKENS.fontMono}`;
   599	          c.fillText(String(opts.centerSubText), cx, cy + 12);
   600	        }
   601	        c.restore();
   602	      },
   603	    });
   604	  }
   605	
   606	  return createDonutChart(canvasId, {
   607	    data: {
   608	      labels: chartItems.map(d => d.label),

   622	        legend: createLegendOptions("hidden"),
   623	        tooltip: createTooltipOptions({

   625	            label: (item) => {

   628	            },
   629	          },

    19	def get_cache_key(screen: str) -> str:
    20	    return f"dashboard:kpis:{screen}"
    21	
    22	
    23	def get_fallback_data(screen: str, repo: DashboardRepository) -> list[KPIInput]:
    24	    if screen == "worker":
    25	        return repo.get_recent_worker_kpis()
    26	    if screen == "promo":
    27	        return repo.get_recent_promo_kpis()
    28	    return []
    29	
    30	
    31	def deserialize_kpis(raw_items: list[dict]) -> list[KPIInput]:
    32	    return [KPIInput.model_validate(item) for item in raw_items]
    33	
    34	
    35	def aggregate_kpi_data(
    36	    screen: str,
    37	    repo: DashboardRepository,
    38	    cache: DashboardCache,
    39	) -> DashboardKPIResponse:
    40	    cache_key = get_cache_key(screen)
    41	    cached = cache.get_json(cache_key)
    42	
    43	    if cached and "items" in cached:
    44	        raw_items = deserialize_kpis(cached["items"])
    45	        refresh_interval = int(cached.get("refresh_interval", DEFAULT_REFRESH_INTERVAL))
    46	    else:
    47	        raw_items = get_fallback_data(screen, repo)
    48	        refresh_interval = DEFAULT_REFRESH_INTERVAL
    49	
    50	    evaluated_items = [
    51	        evaluate_kpi(item, refresh_interval=refresh_interval)
    52	        for item in raw_items
    53	    ]
    54	
    55	    return DashboardKPIResponse(
    56	        screen=screen,
    57	        screenId=SCREEN_ID_MAP[screen],
    58	        refreshInterval=refresh_interval,
    59	        items=evaluated_items,
    60	    )
    61

    63	def get_dashboard_kpis(
    64	    screen: str,
    65	    repo: DashboardRepository | None = None,
    66	    cache: DashboardCache | None = None,
    67	) -> DashboardKPIResponse:
    68	    repo = repo or DashboardRepository()
    69	    cache = cache or DashboardCache()

    72	        raise HTTPException(status_code=400, detail=f"Unsupported screen: {screen}")
