/**
 * 내부 상태값(value) → tone(normal/warning/danger/neutral) 반환
 * @param {string|number} value  - 상태값 또는 수치
 * @param {object} [opts]
 *   opts.thresholds  - [{max, tone}] 오름차순. 미지정 시 기본 규칙 사용
 *   opts.invert      - true: 낮을수록 danger (자재 잔량 등)
 */
function resolveTone(value, opts = {}) {
  // 문자열 상태값 매핑
  const STATUS_TONE = {
    RUN:    "normal",
    IDLE:   "warning",
    DOWN:   "danger",
    normal: "normal",
    warning:"warning",
    danger: "danger",
  };
  if (typeof value === "string") return STATUS_TONE[value] ?? "neutral";

  // 수치 기반 임계치
  const num = Number(value);
  const thresholds = opts.thresholds ?? [
    { max: 39,  tone: opts.invert ? "normal"  : "danger"  },
    { max: 69,  tone: "warning" },
    { max: 100, tone: opts.invert ? "danger"  : "normal"  },
  ];
  for (const t of thresholds) {
    if (num <= t.max) return t.tone;
  }
  return opts.invert ? "danger" : "normal";
}

/**
 * 상태 배지 HTML 반환 (재사용 가능)
 * @param {object} p
 *   p.value      - 내부 상태값 (예: "RUN", "normal", "danger")
 *   p.label      - 표시 텍스트 (예: "가동중", "정상", "위험")
 *   p.tone       - 직접 지정 시 value 해석 생략 ("normal"|"warning"|"danger"|"neutral")
 *   p.showDot    - 점 인디케이터 표시 여부 (기본 true)
 */
function statusBadgeHTML(p) {

    : "";
  return `<span class="status-badge status-badge--${tone}" role="status" aria-label="${esc(p.label)}">${dot}${esc(p.label)}</span>`;
}

/** rawTimestamp 우선 최신순 정렬 (fallback: time 문자열) */
function sortRecentLogs(logs = []) {
  const toMs = (r) => {
    const ts = new Date(r?.rawTimestamp).getTime();
    return Number.isFinite(ts) ? ts : Date.parse(`1970-01-01T${r?.time ?? "00:00:00"}+09:00`);
  };
  return [...logs].sort((a, b) => toMs(b) - toMs(a));
}

function compareDetailValues(a, b, type = "text", direction = "asc") {
  const dir = direction === "desc" ? -1 : 1;

  if (type === "datetime") {
    const aTime = new Date(a).getTime();
    const bTime = new Date(b).getTime();
    const safeA = Number.isFinite(aTime) ? aTime : -Infinity;
    const safeB = Number.isFinite(bTime) ? bTime : -Infinity;
    return (safeA - safeB) * dir;
  }

  if (type === "number" || type === "percent") {
    const safeA = Number.isFinite(Number(a)) ? Number(a) : -Infinity;
    const safeB = Number.isFinite(Number(b)) ? Number(b) : -Infinity;
    return (safeA - safeB) * dir;
  }

  return String(a ?? "").localeCompare(String(b ?? ""), "ko-KR") * dir;
}

function sortDetailRows(rows = [], sortBy = {}) {
  const { key, type = "text", direction = "asc" } = sortBy;
  if (!key) return [...rows];

  return [...rows].sort((a, b) => compareDetailValues(a?.[key], b?.[key], type, direction));
}

function formatDetailCell(row, column) {
  const raw = row?.[column.key];

  if (column.type === "status") {
    const meta = resolveDetailStatusMeta(raw);
    return statusBadgeHTML({ tone: meta.tone, label: meta.label });
  }

  if (raw === null || raw === undefined || raw === "") {
    return `<span class="detail-table-empty">-</span>`;
  }

  if (column.type === "datetime") {
    const display = formatDateTimeDisplay(raw);
    return `<span class="detail-table-mono" title="${esc(display)}">${esc(display)}</span>`;
  }

  if (column.type === "id") {
    return `<span class="detail-table-mono" title="${esc(String(raw))}">${esc(String(raw))}</span>`;
  }

  if (column.type === "number") {
    return `<span class="detail-table-num">${fmt(raw)}</span>`;
  }

  if (column.type === "percent") {
    return `<span class="detail-table-num">${pct(raw)}</span>`;
  }

  const text = String(raw);
  return `<span class="detail-table-text" title="${esc(text)}">${esc(text)}</span>`;
}

function renderDetailTable({ title, columns = [], rows = [], sortBy, emptyText = "상세 데이터 없음" }) {
  const sortedRows = sortDetailRows(rows, sortBy);

  if (!sortedRows.length) {

        <div class="detail-table-empty-state">${esc(emptyText)}</div>

  }

  const bodyHTML = sortedRows.map((row) => {
    const cells = columns.map((column) => {
      const alignClass = column.align === "right" ? "detail-table-td--right" : "";
      const cellClass = column.type ? `detail-table-td--${column.type}` : "";
      return `<td class="${alignClass} ${cellClass}">${formatDetailCell(row, column)}</td>`;
    }).join("");
    return `<tr>${cells}</tr>`;
  }).join("");

  return `
    <div class="card">
      <div class="section-title">${esc(title)}</div>

/**
 * 프로그레스 바 블록 HTML 반환 (재사용 가능)
 * @param {object} p
 *   p.label         - 항목 라벨
 *   p.value         - 현재 값 (숫자)
 *   p.max           - 최대 값 (기본 100)
 *   p.unit          - 단위 (기본 "%")
 *   p.helperText    - 보조 텍스트
 *   p.tone          - 직접 tone 지정 시 임계치 계산 생략
 *   p.invert        - true: 낮을수록 danger
 *   p.thresholds    - 사용자 정의 임계치 배열
 */
function progressBarHTML(p) {
  const max    = p.max  ?? 100;
  const unit   = p.unit ?? "%";
  const pct    = Math.min(Math.max((p.value / max) * 100, 0), 100).toFixed(1);
  const tone   = p.tone ?? resolveTone(p.value, {
    invert:     p.invert,
    thresholds: p.thresholds,
  });
  const displayVal = typeof p.value === "number"
    ? (Number.isInteger(p.value) ? p.value : p.value.toFixed(1))
    : p.value;
  const helperHTML = p.helperText
    ? `<div class="progress-block-helper">${esc(p.helperText)}</div>`
    : "";

  // 색상 강조: 값 색상도 tone 연동
  const valColor = KPI_STATUS_COLOR[tone] ?? KPI_STATUS_COLOR.neutral;

  return `
    <div class="progress-block">
      <div class="progress-block-header">
        <span class="progress-block-label">${esc(p.label)}</span>
        <span class="progress-block-value" style="color:${valColor}">${esc(displayVal)}${esc(unit)}</span>
      </div>
      <div class="progress-block-track" role="progressbar"
           aria-valuenow="${p.value}" aria-valuemin="0" aria-valuemax="${max}"
           aria-label="${esc(p.label)} ${esc(displayVal)}${esc(unit)}">
        <div class="progress-block-fill progress-block-fill--${tone}" style="width:${pct}%"></div>
      </div>
      ${helperHTML}
    </div>`;
}

function kpiCardHTML(p) {

  const statusKey = p.status ?? "neutral";

  if (typeof p.value === "number") {
    displayValue = Number.isInteger(p.value)
      ? p.value.toLocaleString("ko-KR")
      : p.value.toFixed(1);
  }

    </span>` : "";

  const trendHTML = p.trendValue ? `
    <span class="kpi-trend" style="color:${KPI_TREND_COLOR[trendKey]}">

function isExpiredMetric(metric) {
  if (!metric?.validUntil) return false;
  const t = new Date(metric.validUntil).getTime();
  return Number.isFinite(t) && Date.now() > t;
}

function isRenderableMetric(metric) {
  if (!metric) return false;
  if (metric.sourceType !== "simulated") return true;
  return Boolean(metric.validUntil && metric.modelVersion && metric.assumption && metric.caution);
}

function activateTab(tabId) {
  if (currentTab === "promo") {
    promoTimers.forEach(clearInterval);
    promoTimers = [];
  }

  currentTab = tabId;

  document.querySelectorAll(".tab-btn").forEach((btn) => {
    const isActive = btn.dataset.tab === tabId;
    btn.classList.toggle("active", isActive);
    if (isActive) {
      const tabDef = TABS.find((t) => t.id === tabId);
      btn.style.setProperty("--tab-color", tabDef?.color ?? "#22d3a0");
    }
  });

  const content = document.getElementById("content");
  const vm = cache?.viewModel?.[tabId];

  if (!vm) {
    content.innerHTML = `<div class="list-empty">viewModel 없음</div>`;
    return;
  }

  const raw = {
    summary: cache.summary,
    lines: cache.lines,
    defects: cache.defects,
    hourly: cache.hourly,
    ngList: cache.ngList,
    statusData: cache.statusData,
  };

  if (tabId === "worker") content.innerHTML = renderWorkerFromViewModel(vm, raw);
  else if (tabId === "qa") content.innerHTML = renderQAFromViewModel(vm, raw);
  else if (tabId === "manager") content.innerHTML = renderManagerFromViewModel(vm);
  else if (tabId === "promo") content.innerHTML = renderPromoFromViewModel(vm, raw);
  else content.innerHTML = `<div class="list-empty">탭 데이터 없음</div>`;

  requestAnimationFrame(() => {
    if (tabId === "worker") renderWorkerTopDefectBarChart("workerTopDefectBarChart", cache.defects);
    if (tabId === "qa") {
      renderDefectRateChart("qaLineChart", cache.hourly);
      renderDefectBarChart("qaBarChart", cache.defects);
    }
    if (tabId === "manager") {
      if (typeof renderHourlyChartWithSimulatedOverlay === "function") {
        renderHourlyChartWithSimulatedOverlay("managerHourlyChart", cache.hourly, vm.simulated || []);
      } else {
        renderHourlyChart("managerHourlyChart", cache.hourly);
      }
      renderLineOutputChart("managerLineBarChart", cache.lines);
    }
    if (tabId === "promo") {
      renderPromoLineChart("promoLineChart", cache.hourly);
      startPromoClockOnly();
    }
  });
}

function formatChartNumber(value) {
  const num = Number(value);
  if (!Number.isFinite(num)) return "-";
  return num.toLocaleString("ko-KR");
}

function formatChartPercent(value, decimals = 1) {

function formatChartCompactTick(value) {
  const num = Number(value);
  if (!Number.isFinite(num)) return "";
  if (Math.abs(num) >= 1000) return `${(num / 1000).toFixed(1)}k`;
  return Number.isInteger(num) ? String(num) : String(num);
}

function formatChartTimeLabel(value) {
  if (!value) return "-";
  const text = String(value);
  if (/^\d{2}:\d{2}(:\d{2})?$/.test(text)) return text.slice(0, 5);

  const date = new Date(value);
  if (!Number.isFinite(date.getTime())) return text;

  const hh = String(date.getHours()).padStart(2, "0");
  const mm = String(date.getMinutes()).padStart(2, "0");
  return `${hh}:${mm}`;
}

function formatChartDateTime(value) {
  if (!value) return "-";
  const date = new Date(value);
  if (!Number.isFinite(date.getTime())) return String(value);

  const yyyy = date.getFullYear();
  const mm = String(date.getMonth() + 1).padStart(2, "0");
  const dd = String(date.getDate()).padStart(2, "0");
  const hh = String(date.getHours()).padStart(2, "0");
  const mi = String(date.getMinutes()).padStart(2, "0");
  const ss = String(date.getSeconds()).padStart(2, "0");
  return `${yyyy}-${mm}-${dd} ${hh}:${mi}:${ss}`;
}

      font: { family: CHART_TOKENS.fontMono, size: 11 },
      callback: (value) => {
        const txt = formatChartPercent(value, 1);
        return txt === "-" ? "-" : `${txt}%`;
      },
    },
    title: opts.title
      ? {
          display: true,

  };

function prepareDonutDisplayData(items, opts = {}) {
  const maxItems = opts.maxItems ?? 6;
  let chartItems = [...(items ?? [])].sort((a, b) => b.value - a.value);

  if (chartItems.length > maxItems) {
    const topItems = chartItems.slice(0, maxItems);
    const otherValue = chartItems.slice(maxItems).reduce((sum, item) => sum + (Number(item.value) || 0), 0);
    topItems.push({ label: "기타", value: otherValue, color: CHART_TOKENS.donutOther });
    chartItems = topItems;
  }

  const total = chartItems.reduce((sum, item) => sum + (Number(item.value) || 0), 0);
  const legendItems = chartItems.map((item) => ({
    ...item,
    percentText: total ? `${((Number(item.value) || 0) / total * 100).toFixed(1)}%` : "",
    valueText: formatChartNumber(item.value),
  }));

  return { chartItems, legendItems, total };
}

/**

 * @param {string} canvasId
 * @param {Array}  items   - [{ label, value, color }]
 * @param {object} [opts]
 *   opts.centerText     - 중심 큰 텍스트 (null/undefined 이면 숨김)
 *   opts.centerSubText  - 중심 하단 보조 텍스트
 *   opts.maxItems       - 최대 항목 수, 초과 시 "기타" 합산 (기본 6)
 *   opts.cutout         - 도넛 두께 (기본 "62%")
 */
function renderDonutChart(canvasId, items, opts = {}) {
  const cutout = opts.cutout ?? "62%";
  const { chartItems, total } = prepareDonutDisplayData(items, opts);

  // 중심 수치 인라인 플러그인 (opts.centerText 있을 때만 등록)
  const plugins = [];
  if (opts.centerText != null) {
    plugins.push({
      id: "donutCenter",
      beforeDraw(chart) {
        const { ctx: c, chartArea } = chart;
        const cx = chartArea.left + chartArea.width  / 2;
        const cy = chartArea.top  + chartArea.height / 2;
        const hasSub = Boolean(opts.centerSubText);
        c.save();
        c.textAlign    = "center";
        c.textBaseline = "middle";
        c.fillStyle    = CHART_TOKENS.textMain;
        c.font         = `700 16px ${CHART_TOKENS.fontMono}`;
        c.fillText(String(opts.centerText), cx, cy - (hasSub ? 9 : 0));
        if (hasSub) {
          c.fillStyle = CHART_TOKENS.textLabel;
          c.font      = `400 10px ${CHART_TOKENS.fontMono}`;
          c.fillText(String(opts.centerSubText), cx, cy + 12);
        }
        c.restore();
      },
    });
  }

  return createDonutChart(canvasId, {
    data: {
      labels: chartItems.map(d => d.label),
      datasets: [{
        data:             chartItems.map(d => d.value),
        backgroundColor:  chartItems.map(d => d.color),
        borderColor:      CHART_TOKENS.cardBg,
        borderWidth:      2,
        hoverBorderWidth: 3,
      }],
    },
    options: {
      responsive:          true,
      maintainAspectRatio: false,
      cutout,
      plugins: {
        legend: createLegendOptions("hidden"),
        tooltip: createTooltipOptions({
          callbacks: {
            label: (item) => {
              const ratio = total ? (Number(item.raw) / total) * 100 : 0;
              return ` ${item.label}: ${formatChartNumber(item.raw)} (${ratio.toFixed(1)}%)`;
            },
          },
        }),
      },
    },
    plugins,
  });
}

# server_api/src/modules/dashboard/service.py

from fastapi import HTTPException

from src.modules.dashboard.cache import DashboardCache
from src.modules.dashboard.constants import DEFAULT_REFRESH_INTERVAL, SCREEN_ID_MAP
from src.modules.dashboard.evaluator import evaluate_kpi
from src.modules.dashboard.repository import DashboardRepository
from src.modules.dashboard.schemas import DashboardKPIResponse, KPIInput


def dashboard_status_placeholder() -> dict:
    return {
        "module": "dashboard",
        "message": "Dashboard KPI polling service is ready.",
    }


def get_cache_key(screen: str) -> str:
    return f"dashboard:kpis:{screen}"


def get_fallback_data(screen: str, repo: DashboardRepository) -> list[KPIInput]:
    if screen == "worker":
        return repo.get_recent_worker_kpis()
    if screen == "promo":
        return repo.get_recent_promo_kpis()
    return []


def deserialize_kpis(raw_items: list[dict]) -> list[KPIInput]:
    return [KPIInput.model_validate(item) for item in raw_items]


def aggregate_kpi_data(
    screen: str,
    repo: DashboardRepository,
    cache: DashboardCache,
) -> DashboardKPIResponse:
    cache_key = get_cache_key(screen)
    cached = cache.get_json(cache_key)

    if cached and "items" in cached:
        raw_items = deserialize_kpis(cached["items"])
        refresh_interval = int(cached.get("refresh_interval", DEFAULT_REFRESH_INTERVAL))
    else:
        raw_items = get_fallback_data(screen, repo)
        refresh_interval = DEFAULT_REFRESH_INTERVAL

    evaluated_items = [
        evaluate_kpi(item, refresh_interval=refresh_interval)
        for item in raw_items
    ]

    return DashboardKPIResponse(
        screen=screen,
        screenId=SCREEN_ID_MAP[screen],
        refreshInterval=refresh_interval,
        items=evaluated_items,
    )


def get_dashboard_kpis(
    screen: str,
    repo: DashboardRepository | None = None,
    cache: DashboardCache | None = None,
) -> DashboardKPIResponse:
    repo = repo or DashboardRepository()
    cache = cache or DashboardCache()

    if screen not in SCREEN_ID_MAP:
        raise HTTPException(status_code=400, detail=f"Unsupported screen: {screen}")

    return aggregate_kpi_data(screen=screen, repo=repo, cache=cache)
