   116	  if (typeof value === "string") return STATUS_TONE[value] ?? "neutral";
   117	
   118	  // 수치 기반 임계치
   119	  const num = Number(value);
   120	  const thresholds = opts.thresholds ?? [
   121	    { max: 39,  tone: opts.invert ? "normal"  : "danger"  },
   122	    { max: 69,  tone: "warning" },
   123	    { max: 100, tone: opts.invert ? "danger"  : "normal"  },
   124	  ];
   125	  for (const t of thresholds) {
   126	    if (num <= t.max) return t.tone;
   127	  }
   128	  return opts.invert ? "danger" : "normal";
   129	}

   219	function compareDetailValues(a, b, type = "text", direction = "asc") {
   220	  const dir = direction === "desc" ? -1 : 1;
   221	
   222	  if (type === "datetime") {
   223	    const aTime = new Date(a).getTime();
   224	    const bTime = new Date(b).getTime();
   225	    const safeA = Number.isFinite(aTime) ? aTime : -Infinity;
   226	    const safeB = Number.isFinite(bTime) ? bTime : -Infinity;
   227	    return (safeA - safeB) * dir;
   228	  }
   229	
   230	  if (type === "number" || type === "percent") {
   231	    const safeA = Number.isFinite(Number(a)) ? Number(a) : -Infinity;
   232	    const safeB = Number.isFinite(Number(b)) ? Number(b) : -Infinity;
   233	    return (safeA - safeB) * dir;
   234	  }
   235	
   236	  return String(a ?? "").localeCompare(String(b ?? ""), "ko-KR") * dir;
   237	}

   322	 *   p.value         - 현재 값 (숫자)
   323	 *   p.max           - 최대 값 (기본 100)
   324	 *   p.unit          - 단위 (기본 "%")
   325	 *   p.helperText    - 보조 텍스트

   327	 *   p.invert        - true: 낮을수록 danger
   328	 *   p.thresholds    - 사용자 정의 임계치 배열
   329	 */
   330	function progressBarHTML(p) {
   331	  const max    = p.max  ?? 100;
   332	  const unit   = p.unit ?? "%";
   333	  const pct    = Math.min(Math.max((p.value / max) * 100, 0), 100).toFixed(1);
   334	  const tone   = p.tone ?? resolveTone(p.value, {
   335	    invert:     p.invert,
   336	    thresholds: p.thresholds,
   337	  });
   338	  const displayVal = typeof p.value === "number"
   339	    ? (Number.isInteger(p.value) ? p.value : p.value.toFixed(1))
   340	    : p.value;
   341	  const helperHTML = p.helperText
   342	    ? `<div class="progress-block-helper">${esc(p.helperText)}</div>`
   343	    : "";
   344	
   345	  // 색상 강조: 값 색상도 tone 연동
   346	  const valColor = KPI_STATUS_COLOR[tone] ?? KPI_STATUS_COLOR.neutral;
   347	
   348	  return `
   349	    <div class="progress-block">
   350	      <div class="progress-block-header">
   351	        <span class="progress-block-label">${esc(p.label)}</span>
   352	        <span class="progress-block-value" style="color:${valColor}">${esc(displayVal)}${esc(unit)}</span>
   353	      </div>
   354	      <div class="progress-block-track" role="progressbar"
   355	           aria-valuenow="${p.value}" aria-valuemin="0" aria-valuemax="${max}"
   356	           aria-label="${esc(p.label)} ${esc(displayVal)}${esc(unit)}">
   357	        <div class="progress-block-fill progress-block-fill--${tone}" style="width:${pct}%"></div>
   358	      </div>
   359	      ${helperHTML}
   360	    </div>`;
   361	}

   802	/**
   803	 * 홍보 스크린의 실시간 시계 + TOP Defect 롤링 시작
   804	 * @param {Array} defects
   805	 */
   806	function startPromoTimers(defects) {
   807	  // 기존 타이머 정리
   808	  promoTimers.forEach(clearInterval);
   809	  promoTimers = [];
   810	
   811	  // 발생 건수 내림차순 정렬
   812	  const sorted = [...defects].sort((a, b) => b.count - a.count).slice(0, 3);
   813	  let idx = 0;
   814	
   815	  /** TOP Defect 패널 업데이트 */
   816	  function updateTopDefect() {
   817	    const d     = sorted[idx];
   818	    const panel = document.getElementById("topDefectPanel");
   819	    const rank  = document.getElementById("topDefectRank");
   820	    const name  = document.getElementById("topDefectName");
   821	    const count = document.getElementById("topDefectCount");
   822	    const dots  = document.getElementById("topDefectDots");
   823	    if (!panel) return;
   824	
   825	    panel.style.borderColor  = d.color + "44";
   826	    panel.style.boxShadow    = `0 0 40px ${d.color}18`;
   827	    rank.style.background    = d.color + "18";
   828	    rank.style.border        = `2px solid ${d.color}66`;
   829	    rank.textContent         = `#${idx + 1}`;
   830	    name.style.color         = d.color;
   831	    name.textContent         = d.class_name.toUpperCase();
   832	    count.textContent        = `${d.count}건 발생`;
   833	
   834	    // 페이지 인디케이터 점 렌더링
   835	    dots.innerHTML = sorted.map((s, i) =>
   836	      `<div class="dot ${i === idx ? "active" : ""}" style="--dot-color:${s.color}"></div>`
   837	    ).join("");
   838	
   839	    idx = (idx + 1) % sorted.length;
   840	  }
   841	
   842	  updateTopDefect(); // 즉시 1회 실행
   843	  promoTimers.push(setInterval(updateTopDefect, 3000)); // 3초마다 전환
   844	
   845	  /** 실시간 시계 업데이트 */
   846	  function updateClock() {
   847	    const el = document.getElementById("promoClock");
   848	    if (el) el.textContent = new Date().toLocaleTimeString("ko-KR");
   849	  }
   850	
   851	  updateClock();
   852	  promoTimers.push(setInterval(updateClock, 1000)); // 1초마다 갱신
   853	}

   859	/**

   863	function activateTab(tabId) {
   864	  // 홍보 스크린 타이머 정리 (다른 탭으로 이동 시)
   865	  if (currentTab === "promo") {
   866	    promoTimers.forEach(clearInterval);
   867	    promoTimers = [];
   868	  }
   869	
   870	  currentTab = tabId;
   871	
   872	  // 탭 버튼 활성 스타일 적용
   873	  document.querySelectorAll(".tab-btn").forEach(btn => {
   874	    const isActive = btn.dataset.tab === tabId;
   875	    btn.classList.toggle("active", isActive);
   876	    if (isActive) {
   877	      const tabDef = TABS.find(t => t.id === tabId);
   878	      btn.style.setProperty("--tab-color", tabDef?.color ?? "#22d3a0");
   879	    }
   880	  });
   881	
   882	  // 캐시 데이터로 화면 렌더링
   883	  const {
   884	    lines,
   885	    defects,
   886	    hourly,
   887	    ngList,
   888	    recentLogs,
   889	    summary,
   890	    inspectionResult,
   891	    inspectionResults,
   892	    lineProductionStatus,
   893	  } = cache;
   894	  const content = document.getElementById("content");
   895	
   896	  // 화면별 HTML 렌더링
   897	  switch (tabId) {
   898	    case "worker":
   899	      content.innerHTML = renderWorker({ lines, defects, ngList, recentLogs, summary });
   900	      break;
   901	    case "qa":
   902	      content.innerHTML = renderQA({ defects, hourly, inspectionResults });
   903	      break;
   904	    case "manager":
   905	      content.innerHTML = renderManager({ lines, defects, hourly, summary, lineProductionStatus });
   906	      break;
   907	    case "promo":
   908	      content.innerHTML = renderPromo({ defects, hourly });
   909	      break;
   910	  }
   911	
   912	  // 화면별 차트 초기화 (DOM 삽입 후 실행됨)
   913	  requestAnimationFrame(() => {
   914	    switch (tabId) {
   915	      case "worker": {
   916	        renderWorkerLineChart("workerLineChart", hourly);
   917	        // 검사 결과 비율 도넛 (9.5.3.7)
   918	        const inspRawItems = inspectionResult ?? [];
   919	        const { legendItems: inspLegendItems, total: inspTotal } = prepareDonutDisplayData(inspRawItems);
   920	        const yieldItem = inspRawItems.find((d) => d.label === "양품");
   921	        const yieldPct  = (inspTotal && yieldItem)
   922	          ? `${(((Number(yieldItem.value) || 0) / inspTotal) * 100).toFixed(1)}%`
   923	          : "-";
   924	        renderDonutChart("workerInspectionDonut", inspRawItems, {
   925	          centerText:    yieldPct,
   926	          centerSubText: "양품률",
   927	        });
   928	        const wLegend = document.getElementById("workerInspectionLegend");
   929	        if (wLegend) wLegend.innerHTML = donutLegendHTML(inspLegendItems, inspTotal);
   930	        break;
   931	      }
   932	      case "qa":
   933	        renderDefectRateChart("qaLineChart", hourly);
   934	        renderDefectPieChart("qaPieChart", defects);
   935	        renderDefectBarChart("qaBarChart", defects);        // 9.5.3.6
   936	        break;
   937	      case "manager": {
   938	        renderHourlyChart("managerHourlyChart", hourly);
   939	        renderLineOutputChart("managerLineBarChart", lines); // 9.5.3.6
   940	        // 라인 상태 분포 도넛 (9.5.3.7)
   941	        const statusRawItems = buildStatusDistribution(lines);
   942	        const { legendItems: statusLegendItems, total: statusTotal } = prepareDonutDisplayData(statusRawItems);
   943	        renderDonutChart("managerStatusDonut", statusRawItems, {
   944	          centerText:    formatChartNumber(statusTotal),
   945	          centerSubText: "전체 라인",
   946	        });
   947	        const mLegend = document.getElementById("managerStatusLegend");
   948	        if (mLegend) mLegend.innerHTML = donutLegendHTML(statusLegendItems, statusTotal);
   949	        break;
   950	      }
   951	      case "promo":
   952	        renderPromoLineChart("promoLineChart", hourly);
   953	        startPromoTimers(defects);
   954	        break;
   955	    }
   956	  });

    62	function formatChartTimeLabel(value) {
    63	  if (!value) return "-";
    64	  const text = String(value);
    65	  if (/^\d{2}:\d{2}(:\d{2})?$/.test(text)) return text.slice(0, 5);
    66	
    67	  const date = new Date(value);
    68	  if (!Number.isFinite(date.getTime())) return text;

    70	  const hh = String(date.getHours()).padStart(2, "0");
    71	  const mm = String(date.getMinutes()).padStart(2, "0");

    79

   221	  const maxItems = opts.maxItems ?? 6;
   222	  let chartItems = [...(items ?? [])].sort((a, b) => b.value - a.value);

   224	  if (chartItems.length > maxItems) {

   228	    chartItems = topItems;
   229	  }

   232	  const legendItems = chartItems.map((item) => ({
   233	    ...item,

   646	/**
   647	 * 범용 도넛차트 (비중/구성비)
   648	 * @param {string} canvasId
   649	 * @param {Array}  items   - [{ label, value, color }]
   650	 * @param {object} [opts]
   651	 *   opts.centerText     - 중심 큰 텍스트 (null/undefined 이면 숨김)
   652	 *   opts.centerSubText  - 중심 하단 보조 텍스트
   653	 *   opts.maxItems       - 최대 항목 수, 초과 시 "기타" 합산 (기본 6)
   654	 *   opts.cutout         - 도넛 두께 (기본 "62%")
   655	 */
   656	function renderDonutChart(canvasId, items, opts = {}) {
   657	  const cutout = opts.cutout ?? "62%";
   658	  const { chartItems, total } = prepareDonutDisplayData(items, opts);
   659	
   660	  // 중심 수치 인라인 플러그인 (opts.centerText 있을 때만 등록)
   661	  const plugins = [];
   662	  if (opts.centerText != null) {
   663	    plugins.push({
   664	      id: "donutCenter",
   665	      beforeDraw(chart) {
   666	        const { ctx: c, chartArea } = chart;
   667	        const cx = chartArea.left + chartArea.width  / 2;
   668	        const cy = chartArea.top  + chartArea.height / 2;
   669	        const hasSub = Boolean(opts.centerSubText);
   670	        c.save();
   671	        c.textAlign    = "center";
   672	        c.textBaseline = "middle";
   673	        c.fillStyle    = CHART_TOKENS.textMain;
   674	        c.font         = `700 16px ${CHART_TOKENS.fontMono}`;
   675	        c.fillText(String(opts.centerText), cx, cy - (hasSub ? 9 : 0));
   676	        if (hasSub) {
   677	          c.fillStyle = CHART_TOKENS.textLabel;
   678	          c.font      = `400 10px ${CHART_TOKENS.fontMono}`;
   679	          c.fillText(String(opts.centerSubText), cx, cy + 12);
   680	        }
   681	        c.restore();
   682	      },
   683	    });
   684	  }
   685	
   686	  return createDonutChart(canvasId, {
   687	    data: {
   688	      labels: chartItems.map(d => d.label),
   689	      datasets: [{
   690	        data:             chartItems.map(d => d.value),
   691	        backgroundColor:  chartItems.map(d => d.color),
   692	        borderColor:      CHART_TOKENS.cardBg,
   693	        borderWidth:      2,
   694	        hoverBorderWidth: 3,
   695	      }],
   696	    },
   697	    options: {
   698	      responsive:          true,
   699	      maintainAspectRatio: false,
   700	      cutout,
   701	      plugins: {
   702	        legend: createLegendOptions("hidden"),
   703	        tooltip: createTooltipOptions({
   704	          callbacks: {
   705	            label: (item) => {
   706	              const ratio = total ? (Number(item.raw) / total) * 100 : 0;
   707	              return ` ${item.label}: ${formatChartNumber(item.raw)} (${ratio.toFixed(1)}%)`;
   708	            },
   709	          },
   710	        }),
   711	      },
   712	    },
   713	    plugins,
   714	  });
   715	}
