/**
 * 내부 상태값(value) → tone(normal/warning/danger/neutral) 반환
 * @param {string|number} value  - 상태값 또는 수치
 * @param {object} [opts]
 *   opts.thresholds  - [{max, tone}] 오름차순. 미지정 시 기본 규칙 사용
 *   opts.invert      - true: 낮을수록 danger (자재 잔량 등)
 */
function resolveTone(value, opts = {}) {
  // 문자열 상태값 매핑
  const STATUS_TONE = {
    RUN:    "normal",
    IDLE:   "warning",
    DOWN:   "danger",
    normal: "normal",
    warning:"warning",
    danger: "danger",
  };
  if (typeof value === "string") return STATUS_TONE[value] ?? "neutral";

  // 수치 기반 임계치
  const num = Number(value);
  const thresholds = opts.thresholds ?? [
    { max: 39,  tone: opts.invert ? "normal"  : "danger"  },
    { max: 69,  tone: "warning" },
    { max: 100, tone: opts.invert ? "danger"  : "normal"  },
  ];
  for (const t of thresholds) {
    if (num <= t.max) return t.tone;
  }
  return opts.invert ? "danger" : "normal";
}

/**
 * 상태 배지 HTML 반환 (재사용 가능)
 * @param {object} p
 *   p.value      - 내부 상태값 (예: "RUN", "normal", "danger")
 *   p.label      - 표시 텍스트 (예: "가동중", "정상", "위험")
 *   p.tone       - 직접 지정 시 value 해석 생략 ("normal"|"warning"|"danger"|"neutral")
 *   p.showDot    - 점 인디케이터 표시 여부 (기본 true)
 */
function statusBadgeHTML(p) {
  const tone = p.tone ?? resolveTone(p.value);
  const dot  = p.showDot !== false
    ? `<span class="status-badge-dot" aria-hidden="true"></span>`
    : "";
  return `<span class="status-badge status-badge--${tone}" role="status" aria-label="${esc(p.label)}">${dot}${esc(p.label)}</span>`;
}

/** rawTimestamp 우선 최신순 정렬 (fallback: time 문자열) */
function sortRecentLogs(logs = []) {
  const toMs = (r) => {
    const ts = new Date(r?.rawTimestamp).getTime();
    return Number.isFinite(ts) ? ts : Date.parse(`1970-01-01T${r?.time ?? "00:00:00"}+09:00`);
  };
  return [...logs].sort((a, b) => toMs(b) - toMs(a));
}

function compareDetailValues(a, b, type = "text", direction = "asc") {
  const dir = direction === "desc" ? -1 : 1;

  if (type === "datetime") {
    const aTime = new Date(a).getTime();
    const bTime = new Date(b).getTime();
    const safeA = Number.isFinite(aTime) ? aTime : -Infinity;
    const safeB = Number.isFinite(bTime) ? bTime : -Infinity;
    return (safeA - safeB) * dir;
  }

  if (type === "number" || type === "percent") {
    const safeA = Number.isFinite(Number(a)) ? Number(a) : -Infinity;
    const safeB = Number.isFinite(Number(b)) ? Number(b) : -Infinity;
    return (safeA - safeB) * dir;
  }

  return String(a ?? "").localeCompare(String(b ?? ""), "ko-KR") * dir;
}

function sortDetailRows(rows = [], sortBy = {}) {
  const { key, type = "text", direction = "asc" } = sortBy;
  if (!key) return [...rows];

  return [...rows].sort((a, b) => compareDetailValues(a?.[key], b?.[key], type, direction));
}

function formatDetailCell(row, column) {
  const raw = row?.[column.key];

  if (column.type === "status") {
    const meta = resolveDetailStatusMeta(raw);
    return statusBadgeHTML({ tone: meta.tone, label: meta.label });
  }

  if (raw === null || raw === undefined || raw === "") {
    return `<span class="detail-table-empty">-</span>`;
  }

  if (column.type === "datetime") {
    const display = formatDateTimeDisplay(raw);
    return `<span class="detail-table-mono" title="${esc(display)}">${esc(display)}</span>`;
  }

  if (column.type === "id") {
    return `<span class="detail-table-mono" title="${esc(String(raw))}">${esc(String(raw))}</span>`;
  }

  if (column.type === "number") {
    return `<span class="detail-table-num">${fmt(raw)}</span>`;
  }

  if (column.type === "percent") {
    return `<span class="detail-table-num">${pct(raw)}</span>`;
  }

  const text = String(raw);
  return `<span class="detail-table-text" title="${esc(text)}">${esc(text)}</span>`;
}

function renderDetailTable({ title, columns = [], rows = [], sortBy, emptyText = "상세 데이터 없음" }) {
  const sortedRows = sortDetailRows(rows, sortBy);

  if (!sortedRows.length) {

      <div class="card">
        <div class="section-title">${esc(title)}</div>
        <div class="detail-table-empty-state">${esc(emptyText)}</div>

  }

  const bodyHTML = sortedRows.map((row) => {
    const cells = columns.map((column) => {
      const alignClass = column.align === "right" ? "detail-table-td--right" : "";
      const cellClass = column.type ? `detail-table-td--${column.type}` : "";
      return `<td class="${alignClass} ${cellClass}">${formatDetailCell(row, column)}</td>`;
    }).join("");
    return `<tr>${cells}</tr>`;
  }).join("");

  return `
    <div class="card">
      <div class="section-title">${esc(title)}</div>

/**
 * 프로그레스 바 블록 HTML 반환 (재사용 가능)
 * @param {object} p
 *   p.label         - 항목 라벨
 *   p.value         - 현재 값 (숫자)
 *   p.max           - 최대 값 (기본 100)
 *   p.unit          - 단위 (기본 "%")
 *   p.helperText    - 보조 텍스트
 *   p.tone          - 직접 tone 지정 시 임계치 계산 생략
 *   p.invert        - true: 낮을수록 danger
 *   p.thresholds    - 사용자 정의 임계치 배열
 */
function progressBarHTML(p) {
  const max    = p.max  ?? 100;
  const unit   = p.unit ?? "%";
  const pct    = Math.min(Math.max((p.value / max) * 100, 0), 100).toFixed(1);
  const tone   = p.tone ?? resolveTone(p.value, {
    invert:     p.invert,
    thresholds: p.thresholds,
  });
  const displayVal = typeof p.value === "number"
    ? (Number.isInteger(p.value) ? p.value : p.value.toFixed(1))
    : p.value;
  const helperHTML = p.helperText
    ? `<div class="progress-block-helper">${esc(p.helperText)}</div>`
    : "";

  // 색상 강조: 값 색상도 tone 연동
  const valColor = KPI_STATUS_COLOR[tone] ?? KPI_STATUS_COLOR.neutral;

  return `
    <div class="progress-block">
      <div class="progress-block-header">
        <span class="progress-block-label">${esc(p.label)}</span>
        <span class="progress-block-value" style="color:${valColor}">${esc(displayVal)}${esc(unit)}</span>
      </div>
      <div class="progress-block-track" role="progressbar"
           aria-valuenow="${p.value}" aria-valuemin="0" aria-valuemax="${max}"
           aria-label="${esc(p.label)} ${esc(displayVal)}${esc(unit)}">
        <div class="progress-block-fill progress-block-fill--${tone}" style="width:${pct}%"></div>
      </div>
      ${helperHTML}
    </div>`;
}

function kpiCardHTML(p) {
  const title = p.title ?? p.label ?? "";
  const statusKey = p.status ?? "neutral";
  const accentClr = p.accent ?? KPI_STATUS_COLOR[statusKey];
  const subText = p.helperText ?? p.sub ?? "";

  let displayValue = p.value;
  if (typeof p.value === "number") {
    displayValue = Number.isInteger(p.value)
      ? p.value.toLocaleString("ko-KR")
      : p.value.toFixed(1);
  }

  const statusBadge = p.statusText ? `
    <span class="kpi-status-badge"
          style="color:${KPI_STATUS_COLOR[statusKey]}; background:${KPI_STATUS_BG[statusKey]}; border:1px solid ${KPI_STATUS_COLOR[statusKey]}44">
      ${esc(p.statusText)}
    </span>` : "";

  const trendKey = p.trendType ?? "flat";
  const trendHTML = p.trendValue ? `
    <span class="kpi-trend" style="color:${KPI_TREND_COLOR[trendKey]}">
      ${KPI_TREND_ICON[trendKey]} ${esc(p.trendValue)}
    </span>` : "";

  const unitHTML = p.unit ? `<span class="kpi-unit">${esc(p.unit)}</span>` : "";

  return `
    <div class="kpi-card" data-kpi="${esc(p.key ?? "")}"
         style="border:1px solid ${accentClr}33; box-shadow:0 0 20px ${accentClr}14;">

function isExpiredMetric(metric) {
  if (!metric?.validUntil) return false;
  const t = new Date(metric.validUntil).getTime();
  return Number.isFinite(t) && Date.now() > t;
}

function isRenderableMetric(metric) {
  if (!metric) return false;
  if (metric.sourceType !== "simulated") return true;
  return Boolean(metric.validUntil && metric.modelVersion && metric.assumption && metric.caution);
}

function activateTab(tabId) {
  if (currentTab === "promo") {
    promoTimers.forEach(clearInterval);
    promoTimers = [];
  }

  currentTab = tabId;

  document.querySelectorAll(".tab-btn").forEach((btn) => {
    const isActive = btn.dataset.tab === tabId;
    btn.classList.toggle("active", isActive);
    if (isActive) {
      const tabDef = TABS.find((t) => t.id === tabId);
      btn.style.setProperty("--tab-color", tabDef?.color ?? "#22d3a0");
    }
  });

  const content = document.getElementById("content");
  const vm = cache?.viewModel?.[tabId];

  if (!vm) {
    content.innerHTML = `<div class="list-empty">viewModel 없음</div>`;
    return;
  }

  const raw = {
    summary: cache.summary,
    lines: cache.lines,
    defects: cache.defects,
    hourly: cache.hourly,
    ngList: cache.ngList,
    statusData: cache.statusData,
  };

  if (tabId === "worker") content.innerHTML = renderWorkerFromViewModel(vm, raw);
  else if (tabId === "qa") content.innerHTML = renderQAFromViewModel(vm, raw);
  else if (tabId === "manager") content.innerHTML = renderManagerFromViewModel(vm);
  else if (tabId === "promo") content.innerHTML = renderPromoFromViewModel(vm, raw);
  else content.innerHTML = `<div class="list-empty">탭 데이터 없음</div>`;

  requestAnimationFrame(() => {
    if (tabId === "worker") renderWorkerTopDefectBarChart("workerTopDefectBarChart", cache.defects);
    if (tabId === "qa") {
      renderDefectRateChart("qaLineChart", cache.hourly);
      renderDefectBarChart("qaBarChart", cache.defects);
    }
    if (tabId === "manager") {
      if (typeof renderHourlyChartWithSimulatedOverlay === "function") {
        renderHourlyChartWithSimulatedOverlay("managerHourlyChart", cache.hourly, vm.simulated || []);
      } else {
        renderHourlyChart("managerHourlyChart", cache.hourly);
      }
      renderLineOutputChart("managerLineBarChart", cache.lines);
    }
    if (tabId === "promo") {
      renderPromoLineChart("promoLineChart", cache.hourly);
      startPromoClockOnly();
    }
  });
}

function formatChartNumber(value) {
  const num = Number(value);
  if (!Number.isFinite(num)) return "-";
  return num.toLocaleString("ko-KR");
}

function formatChartPercent(value, decimals = 1) {
  const num = Number(value);
  if (!Number.isFinite(num)) return "-";
  return num.toFixed(decimals);
}

function formatChartCompactTick(value) {
  const num = Number(value);
  if (!Number.isFinite(num)) return "";
  if (Math.abs(num) >= 1000) return `${(num / 1000).toFixed(1)}k`;
  return Number.isInteger(num) ? String(num) : String(num);
}

function formatChartTimeLabel(value) {
  if (!value) return "-";
  const text = String(value);
  if (/^\d{2}:\d{2}(:\d{2})?$/.test(text)) return text.slice(0, 5);

  const date = new Date(value);
  if (!Number.isFinite(date.getTime())) return text;

  const hh = String(date.getHours()).padStart(2, "0");
  const mm = String(date.getMinutes()).padStart(2, "0");
  return `${hh}:${mm}`;
}

function formatChartDateTime(value) {
  if (!value) return "-";
  const date = new Date(value);
  if (!Number.isFinite(date.getTime())) return String(value);

  const yyyy = date.getFullYear();
  const mm = String(date.getMonth() + 1).padStart(2, "0");
  const dd = String(date.getDate()).padStart(2, "0");
  const hh = String(date.getHours()).padStart(2, "0");
  const mi = String(date.getMinutes()).padStart(2, "0");
  const ss = String(date.getSeconds()).padStart(2, "0");
  return `${yyyy}-${mm}-${dd} ${hh}:${mi}:${ss}`;
}

      font: { family: CHART_TOKENS.fontMono, size: 11 },
      callback: (value) => {
        const txt = formatChartPercent(value, 1);
        return txt === "-" ? "-" : `${txt}%`;
      },
    },
    title: opts.title
      ? {
          display: true,
          text: opts.title,

  };
}

function prepareDonutDisplayData(items, opts = {}) {
  const maxItems = opts.maxItems ?? 6;
  let chartItems = [...(items ?? [])].sort((a, b) => b.value - a.value);

  if (chartItems.length > maxItems) {
    const topItems = chartItems.slice(0, maxItems);
    const otherValue = chartItems.slice(maxItems).reduce((sum, item) => sum + (Number(item.value) || 0), 0);
    topItems.push({ label: "기타", value: otherValue, color: CHART_TOKENS.donutOther });
    chartItems = topItems;
  }

  const total = chartItems.reduce((sum, item) => sum + (Number(item.value) || 0), 0);
  const legendItems = chartItems.map((item) => ({
    ...item,
    percentText: total ? `${((Number(item.value) || 0) / total * 100).toFixed(1)}%` : "",
    valueText: formatChartNumber(item.value),
  }));

  return { chartItems, legendItems, total };
}

/**

 * @param {string} canvasId
 * @param {Array}  items   - [{ label, value, color }]
 * @param {object} [opts]
 *   opts.centerText     - 중심 큰 텍스트 (null/undefined 이면 숨김)
 *   opts.centerSubText  - 중심 하단 보조 텍스트
 *   opts.maxItems       - 최대 항목 수, 초과 시 "기타" 합산 (기본 6)
 *   opts.cutout         - 도넛 두께 (기본 "62%")
 */
function renderDonutChart(canvasId, items, opts = {}) {
  const cutout = opts.cutout ?? "62%";
  const { chartItems, total } = prepareDonutDisplayData(items, opts);

  // 중심 수치 인라인 플러그인 (opts.centerText 있을 때만 등록)
  const plugins = [];
  if (opts.centerText != null) {
    plugins.push({
      id: "donutCenter",
      beforeDraw(chart) {
        const { ctx: c, chartArea } = chart;
        const cx = chartArea.left + chartArea.width  / 2;
        const cy = chartArea.top  + chartArea.height / 2;
        const hasSub = Boolean(opts.centerSubText);
        c.save();
        c.textAlign    = "center";
        c.textBaseline = "middle";
        c.fillStyle    = CHART_TOKENS.textMain;
        c.font         = `700 16px ${CHART_TOKENS.fontMono}`;
        c.fillText(String(opts.centerText), cx, cy - (hasSub ? 9 : 0));
        if (hasSub) {
          c.fillStyle = CHART_TOKENS.textLabel;
          c.font      = `400 10px ${CHART_TOKENS.fontMono}`;
          c.fillText(String(opts.centerSubText), cx, cy + 12);
        }
        c.restore();
      },
    });
  }

  return createDonutChart(canvasId, {
    data: {
      labels: chartItems.map(d => d.label),
      datasets: [{
        data:             chartItems.map(d => d.value),
        backgroundColor:  chartItems.map(d => d.color),
        borderColor:      CHART_TOKENS.cardBg,
        borderWidth:      2,
        hoverBorderWidth: 3,
      }],
    },
    options: {
      responsive:          true,
      maintainAspectRatio: false,
      cutout,
      plugins: {
        legend: createLegendOptions("hidden"),
        tooltip: createTooltipOptions({
          callbacks: {
            label: (item) => {
              const ratio = total ? (Number(item.raw) / total) * 100 : 0;
              return ` ${item.label}: ${formatChartNumber(item.raw)} (${ratio.toFixed(1)}%)`;
            },
          },
        }),
      },
    },
    plugins,
  });
}

function buildSimulatedPredictions(raw, actualMetrics, derivedMetrics) {

# server_api/src/modules/dashboard/service.py

from fastapi import HTTPException

from src.modules.dashboard.cache import DashboardCache
from src.modules.dashboard.constants import DEFAULT_REFRESH_INTERVAL, SCREEN_ID_MAP
from src.modules.dashboard.evaluator import evaluate_kpi
from src.modules.dashboard.repository import DashboardRepository
from src.modules.dashboard.schemas import DashboardKPIResponse, KPIInput


def dashboard_status_placeholder() -> dict:
    return {
        "module": "dashboard",
        "message": "Dashboard KPI polling service is ready.",
    }


def get_cache_key(screen: str) -> str:
    return f"dashboard:kpis:{screen}"


def get_fallback_data(screen: str, repo: DashboardRepository) -> list[KPIInput]:
    if screen == "worker":
        return repo.get_recent_worker_kpis()
    if screen == "promo":
        return repo.get_recent_promo_kpis()
    return []


def deserialize_kpis(raw_items: list[dict]) -> list[KPIInput]:
    return [KPIInput.model_validate(item) for item in raw_items]


def aggregate_kpi_data(
    screen: str,
    repo: DashboardRepository,
    cache: DashboardCache,
) -> DashboardKPIResponse:
    cache_key = get_cache_key(screen)
    cached = cache.get_json(cache_key)

    if cached and "items" in cached:
        raw_items = deserialize_kpis(cached["items"])
        refresh_interval = int(cached.get("refresh_interval", DEFAULT_REFRESH_INTERVAL))
    else:
        raw_items = get_fallback_data(screen, repo)
        refresh_interval = DEFAULT_REFRESH_INTERVAL

    evaluated_items = [
        evaluate_kpi(item, refresh_interval=refresh_interval)
        for item in raw_items
    ]

    return DashboardKPIResponse(
        screen=screen,
        screenId=SCREEN_ID_MAP[screen],
        refreshInterval=refresh_interval,
        items=evaluated_items,
    )


def get_dashboard_kpis(
    screen: str,
    repo: DashboardRepository | None = None,
    cache: DashboardCache | None = None,
) -> DashboardKPIResponse:
    repo = repo or DashboardRepository()
    cache = cache or DashboardCache()

    if screen not in SCREEN_ID_MAP:
        raise HTTPException(status_code=400, detail=f"Unsupported screen: {screen}")

    return aggregate_kpi_data(screen=screen, repo=repo, cache=cache)

# 9.5.1-7 화면별 KPI 최종 구성안

## 핵심 요약
본 문서는 스마트 팩토리 불량 검출 시스템의 각 대시보드 목적에 맞추어 KPI를 화면별로 3~6개 수준으로 압축하고, 역할별 의사결정에 최적화된 최종 KPI 세트를 정의한 결과입니다.  
본 최종 구성안은 **KPI 우선순위, 데이터 가용성, 화면 목적, 시인성, 즉시 대응 필요성**을 함께 고려하여 각 화면에 우선 배치할 KPI 세트를 확정한 기준안입니다.

---

## 1. 개요
화면별 KPI 최종 구성은 단순히 중요한 지표를 많이 배치하는 작업이 아니라,  
**각 역할이 해당 화면에서 가장 먼저 확인해야 하는 질문에 답할 수 있는 지표만 선별하여 상단에 배치하는 작업**입니다.

따라서 본 문서에서는 다음 기준을 적용합니다.

- 각 화면의 존재 목적에 직접 연결되는 KPI를 우선 반영한다.
- Must KPI를 중심으로 구성하되, 필요 시 Should KPI를 보조적으로 포함한다.
- 한 화면의 상단 KPI 카드는 과밀하지 않도록 4개 내외를 원칙으로 한다.
- 동일한 의미의 지표는 화면별 집계 범위만 달리 적용하고 계산 기준은 최대한 통일한다.
- 상세 원인 분석에 적합한 항목은 상단 카드보다 차트 또는 하단 리스트로 배치한다.

---

## 2. 화면별 KPI 최종 구성

### 2.1 현장 대시보드 (SCR-001)

- **목적**: 실시간 상태 모니터링 및 불량 급증 시 즉각 대응
- **대상**: 현장 작업자

현장 대시보드는 즉시 대응이 가장 중요하므로, 작업자가 화면 진입 직후  
**설비 상태 이상 여부**, **불량 급증 여부**, **현재 생산 진척도**를 한눈에 판단할 수 있도록 구성합니다.

| KPI명 | 선정 사유 | 관련 의사결정 | 중요도 |
|---|---|---|---|
| 최근 10분 NG 발생 건수 | 실시간 불량 급증 여부를 가장 빠르게 파악할 수 있음 | 라인 정지 및 긴급 점검 결정 | Must |
| 설비 가동 상태 | 현재 라인의 정상 동작 여부(RUN / DOWN / IDLE)를 즉시 확인 가능 | 즉각적인 설비 복구 조치 | Must |
| 목표 달성률 | 당일 계획 대비 현재 진척도를 직관적으로 확인할 수 있음 | 생산 속도 조절 및 잔업 판단 | Must |
| 누적 생산량 | 현재 작업 실적을 수량 기준으로 빠르게 확인할 수 있음 | 작업 진행 상황 및 생산 집중도 판단 | Must |

#### 구성 원칙
- 현장 화면은 **즉시 대응형 KPI**를 우선 배치합니다.
- `현재 불량률`은 중요하지만 작업자 화면에서는 상단 카드보다 **보조 품질 블록 또는 차트 영역**에 두는 것이 더 적절합니다.

---

### 2.2 품질 대시보드 (SCR-002)

- **목적**: 불량 원인 분석 및 재검/최종 판정 의사결정
- **대상**: 품질관리자(QA)

품질 대시보드는 현재 품질 수준, 주요 불량 원인, 사람이 추가 확인해야 할 대상을 빠르게 파악할 수 있도록 구성합니다.

| KPI명 | 선정 사유 | 관련 의사결정 | 중요도 |
|---|---|---|---|
| 현재 불량률 | 전체 검사 대비 누적 불량 비율을 빠르게 파악할 수 있음 | 품질 목표 달성 여부 판단 | Must |
| 주요 불량 유형 (Top 5) | 가장 빈번한 불량 원인을 식별할 수 있음 | 공정 개선 및 원인 분석 우선순위 결정 | Must |
| 재검 필요 수량 | AI 확신도가 낮아 수동 검토가 필요한 대상을 확인할 수 있음 | 작업 우선순위 및 인력 배분 | Must |
| 검사 현황 요약 (총 검사 수량 / 양품 수 / 불량 수) | 당일 전체 품질 상태를 총량 기준으로 파악할 수 있음 | 전체 품질 실적 보고 및 이상 여부 확인 | Must |
| 예측 불량률 (ML) | 자재/환경 변수 기반 향후 불량 추세를 보조적으로 참고할 수 있음 | 선제적 품질 사고 예방 | Should |

#### 구성 원칙
- 품질 화면은 **현재 상태 + 원인 분석 + 재검 판단**이 동시에 가능해야 합니다.
- `검사 현황 요약`은 단일 KPI가 아니라 **묶음형 요약 카드**로 해석합니다.
- `예측 불량률`은 초기 구현에서는 Should로 두고, 데이터 확보 이후 확대 적용합니다.

---

### 2.3 통합 대시보드 (SCR-003)

- **목적**: 전사 생산 지표 통합 관제 및 운영 자원 배분
- **대상**: 공장 관리자

통합 대시보드는 공장 전체의 효율, 계획 달성 가능성, 라인 간 편차를 함께 판단할 수 있도록 구성합니다.

| KPI명 | 선정 사유 | 관련 의사결정 | 중요도 |
|---|---|---|---|
| 전체 라인 OEE | 가동률, 성능, 품질을 통합한 종합 효율 지표 | 공장 전체 생산성 평가 | Must |
| 공장 전체 목표 달성률 | 전사 계획 대비 현재 실적 진척도를 한눈에 파악 가능 | 전사 생산 계획 조정 | Must |
| 예상 종료 생산량 | 현재 속도 유지 시 종료 시점의 실적을 예측할 수 있음 | 납기 준수 여부 및 자원 재배치 | Must |
| 라인별 생산량 비교 | 라인 간 성과 편차와 병목 구간을 식별할 수 있음 | 병목 해소 및 인력 지원 결정 | Must |
| 불량 위험도 예측 | 향후 단기 품질 리스크를 보조적으로 확인할 수 있음 | 선제적 리스크 관리 | Should |

#### 구성 원칙
- 관리자 화면은 **비교형 KPI**와 **예측형 KPI**를 함께 포함합니다.
- `공장 전체 목표 달성률`은 작업자 화면의 목표 달성률과 구분하기 위해 전사 집계 기준임을 명확히 합니다.

---

### 2.4 생산 현황 보드 (SCR-004)

- **목적**: 핵심 성과 공유 및 현장 공감대 형성
- **대상**: 전 사원 (공용 송출)

생산 현황 보드는 복잡한 분석보다, 공장 전체의 현재 상태를 멀리서도 한눈에 읽을 수 있도록 구성합니다.

| KPI명 | 선정 사유 | 관련 의사결정 | 중요도 |
|---|---|---|---|
| 누적 총 생산량 | 당일 공장 전체 성과를 대형 수치로 직관적으로 전달 | 목표 달성 의지 고취 | Must |
| 목표 달성률 | 계획 대비 현재 위치를 직관적으로 공유 | 생산 진척 상황 인지 | Must |
| 당일 양품률 | 공장 전체 품질 수준을 한눈에 전달 | 품질 우선 문화 형성 | Must |
| TOP 3 Defect Class | 현재 가장 주의해야 할 불량 유형을 요약하여 전파 | 품질 주의사항 현장 공유 | Must |

#### 구성 원칙
- 공용 보드는 **시인성**과 **즉시 이해 가능성**을 최우선으로 합니다.
- 세부 로그나 긴 설명보다는 숫자와 핵심 불량 유형 요약 중심으로 구성합니다.

---

## 3. 공용 KPI와 역할 특화 KPI 구분

시스템의 일관성과 재사용성을 높이기 위해 KPI는 공용 KPI와 역할 특화 KPI로 구분하여 관리합니다.

### 3.1 공용 KPI (Standard KPIs)
공용 KPI는 여러 화면에서 공통으로 사용하되, 화면별 집계 범위만 다르게 적용하는 표준 지표입니다.

- **목표 달성률**  
  - 계산 기준: `해당 화면의 집계 범위(라인/공장)에 대한 현재 양품 수량 ÷ 목표 수량 × 100`
- **현재 불량률**  
  - 계산 기준: `해당 화면의 집계 범위(라인/공장)에 대한 누적 불량 수량 ÷ 누적 검사 수량 × 100`
- **누적 생산량 계열 지표**
  - 작업자 화면: 라인 또는 작업 기준 누적 생산량
  - 관리자/공용 화면: 공장 전체 누적 생산량

### 3.2 역할 특화 KPI (Specialized KPIs)
역할 특화 KPI는 특정 역할의 핵심 의사결정에 직접 연결되는 전용 지표입니다.

- **현장 작업자**
  - 최근 10분 NG 발생 건수
  - 설비 가동 상태

- **품질관리자**
  - 재검 필요 수량
  - 주요 불량 유형 (Top 5)
  - 예측 불량률

- **공장 관리자**
  - 전체 라인 OEE
  - 예상 종료 생산량
  - 라인별 생산량 비교
  - 불량 위험도 예측

- **공용 보드**
  - TOP 3 Defect Class
  - 당일 양품률

---

## 4. 화면별 배치 원칙

### 4.1 상단 KPI 카드
- 작업자 / 품질 / 관리자 대시보드는 **상단 KPI 카드 4개 내외**를 기본으로 합니다.
- 공용 보드는 **대형 KPI 3개 + 보조 정보 1개** 구성을 원칙으로 합니다.
- 상단 KPI는 가능한 한 **즉시 상태 판단이 가능한 지표**를 우선 배치합니다.

### 4.2 중단 차트 영역
- 상단 KPI의 원인 분석이 필요한 항목은 중단 차트로 배치합니다.
- 추세형 KPI는 라인차트, 비교형 KPI는 막대차트, 구성형 KPI는 도넛/막대차트 중심으로 구성합니다.

### 4.3 하단 상세 리스트
- 상세 로그, 불량 상세 목록, 검사 이력, 알람 목록 등은 하단 리스트 또는 테이블로 배치합니다.
- 공용 전시 보드는 가독성을 위해 하단 상세 리스트를 최소화합니다.

---

## 5. 시각화 및 배치 가이드라인

한 화면에 정보가 과밀되지 않도록 다음 원칙을 적용합니다.

- **KPI 카드 (상단)**  
  핵심 수치를 가로 배열하며, 임계값에 따라 배경색 또는 텍스트 색상을 변경하여 상태 인지를 돕습니다.

- **차트 영역 (중단)**  
  KPI의 추세, 비교, 분포를 시각화하여 원인 분석과 상황 이해를 지원합니다.

- **상세 리스트 (하단)**  
  실시간 로그, 검사 대상, 상세 데이터 테이블을 배치하여 후속 확인이 가능하도록 합니다.

- **중복 최소화 원칙**  
  동일한 의미의 KPI를 한 화면 내에서 반복 배치하지 않고, 상단-중단-하단 역할을 구분합니다.

---

## 6. 정리
본 문서는 화면별 KPI 최종 묶음을 확정한 기준 문서로서, 이후 차트 유형 선정, UI 배치, API 응답 구성, 더미데이터 검증의 기준으로 활용됩니다.

따라서 후속 구현 단계에서는 단순히 많은 정보를 넣는 방향이 아니라,  
**각 화면의 목적에 맞는 핵심 KPI를 가장 먼저 보여주고, 나머지 정보는 분석 영역으로 분리하는 방향**을 유지해야 합니다.

즉, 화면별 KPI 최종 구성의 목적은  
**정보 나열이 아니라 역할별 판단 최적화**에 있습니다.

# 9.5.2 최종 차트 유형 선정서_(최종)

## 핵심 요약
본 문서는 스마트 팩토리 불량 검출 시스템의 4개 화면에 대해 KPI별 최종 시각화 유형을 확정하고, 그 선정 근거와 구현 우선순위, 데이터 요구사항 요약을 정리한 최종 기준서이다.  
본 문서는 앞서 수행된 KPI 성격 분류, 차트 후보안 도출, 화면별 배치 초안, 1차/2차 구현 범위 구분, 데이터 요구사항 정의 결과를 통합한 문서로서, 후속 단계인 **9.5.3 차트/테이블 구현**의 직접 기준으로 활용한다.

---

## 1. 작업 목적
최종 차트 유형 선정의 목적은 화면별 KPI에 대해 가장 적합한 시각화 유형을 확정하여, 사용자가 각 화면에서 필요한 판단을 가장 빠르고 정확하게 수행할 수 있도록 구현 기준을 정립하는 데 있다.

세부 목적은 다음과 같다.

- KPI별 최종 차트 유형을 확정한다.
- 1차 구현 대상과 2차 고도화 대상을 구분한다.
- 화면 목적, KPI 성격, 구현 가능성, 데이터 정합성을 종합 반영한다.
- 후속 구현 단계에서 화면 간 일관된 시각화 기준으로 활용한다.

---

## 2. 최종 선정 기준

### 2.1 기본 선정 원칙
최종 차트 유형은 아래 원칙을 기준으로 선정한다.

- **해석 최적화**: KPI의 성격(현황, 상태, 추세, 비교, 분포, 예측, 상세)에 가장 적합한 차트를 우선 적용한다.
- **역할별 위계**: 작업자(즉시 대응), 품질관리자(원인 분석), 관리자(성과 비교 및 운영 판단), 공용 보드(원거리 시인성)에 맞는 표현 구조를 적용한다.
- **구현 현실성**: 1차 구현은 현재 Chart.js 및 HTML/CSS 구조에서 안정적으로 구현 가능한 표준 차트를 우선 적용한다.
- **데이터 정합성**: 차트가 요구하는 최소 데이터 필드와 API 응답 구조를 일치시키는 방향으로 선정한다.

### 2.2 1차 / 2차 구분 원칙
- **1차 구현(Phase 1)**: KPI 카드, 상태 배지, 프로그레스 바 카드, 라인차트, 수평/수직 바차트, 도넛차트, 로그 리스트, 상세 테이블
- **2차 고도화(Phase 2)**: 게이지 차트, 자동 롤링 고도화, 점선형 예측 구간, 카드 애니메이션, 스파크라인 카드, 이미지 썸네일 확장, 복합 차트 고도화

### 2.3 적용 범위 유의사항
- 본 문서의 최종 차트는 **기본 시각화 기준**을 의미한다.
- 경고 배너, 자동 롤링, 점멸 효과, 카운트 업 애니메이션 등은 **보조 UI 요소 또는 고도화 연출 요소**로 관리한다.
- 구현 코드 위치 및 함수명은 참고 자료로 활용하되, 본 문서의 핵심은 **최종 시각화 의사결정 기준**에 둔다.

# 9.5.3 시각화 구현 우선순위 및 데이터 표현 정책서

## 핵심 요약
본 문서는 9.5.1 화면별 KPI 최종 구성안과 9.5.2 최종 차트 유형 선정 결과를 실제 구현 단계로 연결하기 위한 실행 기준서이다.  
목적은 현재 프로젝트의 화면 구조, 코드 구조, 데이터 가용 범위를 기준으로 시각화 구현 우선순위를 확정하고, 각 지표를 `actual / derived / simulated` 데이터로 구분하여 화면 완성도와 데이터 정합성을 동시에 확보하는 데 있다.

본 문서는 다음 5가지를 구현 기준으로 고정한다.

- 현재 프런트 화면 구조 `worker / qa / manager / promo`를 유지한다.
- 1차 구현 범위는 `KPI 카드 / 상태 배지 / 프로그레스 바 / 라인차트 / 바차트 / 도넛차트 / 로그 리스트 / 상세 테이블`을 기본 축으로 한다.
- DB 구조 변경 없이 현재 API 및 집계 데이터로 직접 표현 가능한 항목을 우선 구현한다.
- 부족한 표현 요소는 `actual / derived / simulated` 계층으로 분리하되, `simulated` 지표는 실제 운영 지표와 명확히 구분하여 표시한다.
- 1차 구현에서는 표준형 시각화를 기본으로 하되, 화면 완성도와 2차 구현 수준의 인상을 확보하기 위해 일부 고도화 표현 요소를 경량 방식으로 포함한다.

## 1. 문서 목적

본 문서의 목적은 다음과 같다.

1. 9.5.1에서 확정한 화면별 KPI를 실제 화면 컴포넌트 단위로 변환한다.
2. 9.5.2에서 확정한 차트 유형을 현재 프로젝트 코드 구조에 맞춰 구현 우선순위로 재정렬한다.
3. DB 수정 없이도 화면 품질을 확보할 수 있도록 데이터 표현 정책을 정의한다.
4. 2차 구현처럼 보이는 화면 완성도를 확보하기 위해 경량 고도화 표현의 반영 기준을 정의한다.
5. 후속 단계인 9.5.3.x 세부 구현 작업의 기준 문서로 활용한다.

## 2. 입력 기준 문서 및 구현 전제

### 2.1 입력 기준 문서
본 정책서는 아래 문서를 직접 기준으로 사용한다.

- `9.5.1-7 화면별 KPI 최종 구성안_(최종).md`
- `9.5.2 최종 차트 유형 선정서_(최종).md`
- `조사 계획 보강 및 최종 보고서 구조.md`
- 현재 프런트/백엔드 추출본
  - `web_dashboard.html`
  - `dashboard_app.js`
  - `dashboard_charts.js`
  - `dashboard_api.js`
  - `dashboard_style.css`
  - `service.py`
  - `router.py`
  - DB inventory 추출본

### 2.2 현재 구현 전제
현재 대시보드는 탭 기반 단일 화면 구조로 구성되어 있으며, 작업자(`worker`), 품질관리자(`qa`), 관리자(`manager`), 생산 현황 보드(`promo`) 화면을 각각 별도 렌더 단위로 출력한다.

프런트는 HTML/CSS/JavaScript 기반이며, 차트는 Chart.js 계열의 표준 시각화 중심으로 구현한다.  
따라서 1차 구현에서는 복잡한 엔진 전체를 먼저 완성하기보다, 재사용 가능한 표준 컴포넌트와 안정적인 데이터 바인딩을 기반으로 하고, 필요한 고도화 요소는 `simulation layer`, `rule-based calculation`, `mock inference`, 경량 인터랙션 방식으로 우선 반영한다.

### 2.3 구현 원칙
- 화면 목적에 직접 연결되는 KPI만 상단 우선 영역에 배치한다.
- 분석용 상세 정보는 차트 또는 테이블 영역으로 내린다.
- 1차 구현은 해석이 빠른 표준형 표현을 우선 적용한다.
- 고도화 표현은 허용하되, 실제 엔진 구현과 분리하여 경량 방식으로 반영한다.
- 시뮬레이션 값은 실제 값처럼 보이더라도 `sourceType`과 유효기간 정책을 반드시 포함한다.
- 공통화는 필요한 범위까지만 적용하고, 화면별 목적 차이는 override 가능하게 유지한다.

## 3. 1차 구현 확정 원칙

### 3.1 유지 원칙
- 화면 역할은 `worker / qa / manager / promo`로 고정한다.
- 데이터 표현 계층은 `actual / derived / simulated` 3종으로 고정한다.
- `worker`는 즉시 대응 중심, `qa`는 분석·판정 중심, `manager`는 집계·비교 중심, `promo`는 원거리 가독성 중심으로 유지한다.
- 로그 리스트는 `worker` 우선, 상세 테이블은 `qa` 우선으로 유지한다.
- 1차 구현에서는 화려한 연출보다 해석 속도와 데이터 정합성을 우선한다.
- `simulated` 값은 허용하되 `actual`과 동일한 위계와 강도로 노출하지 않는다.
- 2차 구현 인상을 주는 표현 요소는 `simulated / mock / rule-based` 방식으로 1차에 부분 반영할 수 있다.

### 3.2 교체 원칙
- `현재 불량률`은 1차 구현 기준에서 `derived`로 고정한다.
- `manager` 화면 상단 KPI 수는 기본 3개, 최대 4개 허용으로 고정한다.
- `promo` 1순위는 대형 KPI 3개와 단순 추세 1개를 기본으로 한다.
- 우선순위 표기는 아래 3단계로 통일한다.
  - `P1`: 1차 즉시 구현
  - `P2`: 1차 보조 구현
  - `P3`: 후순위/정교화
- 공통화는 토큰, 포맷터, 차트 기본 옵션, 상태 배지 규칙, 빈 상태 처리 규칙까지로 제한한다.
- 화면별 레이아웃, 문구, 정보 밀도, 차트 조합은 각 화면에서 override 가능하게 둔다.
- `simulated` 값은 카드 제목, 보조 라벨, 배지 중 최소 2개 이상으로 실제값과 시각적으로 분리한다.

### 3.3 추가 원칙
- 데이터 상태 처리 규칙을 명확히 둔다.
  - `null`: `N/A` 또는 `데이터 없음` 표시
  - `0`: 정상 값으로 처리, 숨기지 않음
  - `empty array`: 빈 상태 UI 표시
  - `error`: 오류 문구 또는 fallback block 표시
  - `stale`: 최신 아님 배지 또는 갱신 시각 표시
- 단위 표기 기준을 고정한다.
  - 비율: `%`
  - 개수: `건`
  - 수량: `ea` 또는 `pcs` 중 하나로 문서 전체 통일
  - 시간: `분`
- 시간 기준을 고정한다.
  - `최근 10분`: 현재 시각 기준 직전 10분
  - `당일`: 로컬 날짜 00:00~현재
  - `시간대별`: 당일 시간 버킷 기준
  - `전일 비교`: 전일 동일 시간대 기준
- `derived` 계산 책임은 가능하면 BE/API 우선, 불가 시 FE 계산으로 한다.
- 1차 구현에서 FE 계산 허용 범위는 아래로 제한한다.
  - 비율 계산
  - Top N 집계 재정렬
  - 단순 진행률
  - 현재 속도 기반 예상 종료 생산량
- `simulated` 최소 메타 필드는 아래로 고정한다.
  - `sourceType`
  - `assumption`
  - `validUntil`
  - `modelVersion`
  - `caution`
- 데이터 없음/null/stale 처리는 공통 컴포넌트에서 1차 처리하고, 화면별 호출부는 문구만 override한다.
- 고도화 요소는 실제 데이터 엔진과 분리하여 프런트 표현 계층 또는 simulation layer에서 우선 반영한다.

## 4. 화면별 구현 우선순위 정책

## 4.1 현장 대시보드 (SCR-001 / worker)

### 화면 목적
현장 대시보드는 실시간 상태 모니터링과 즉시 대응이 핵심이다.  
작업자가 화면 진입 직후 설비 상태 이상 여부, 최근 불량 급증 여부, 현재 생산 진척도를 빠르게 판단할 수 있어야 한다.

### 1차 구현 기준
#### P1
- 최근 10분 NG 발생 건수: KPI 카드
- 설비 가동 상태: 상태 배지
- 목표 달성률: 프로그레스 바 카드
- 누적 생산량: KPI 카드
- 실시간 NG 로그: 로그 리스트
- 핵심 KPI 보조 스파크라인
- KPI 숫자 카운트업 애니메이션
- 단순 게이지형 진행률 표현

#### P2
- 최근 발생 유형 Top 3: 수평 바차트
- 보조 품질 블록: 현재 불량률 또는 경고 문구
- 경고 배너 고도화
- 상태 기반 룰형 인사이트 문구

#### P3
- 미니 추세선 고도화
- 애니메이션 세부 정교화
- 예측 경고 표현 확대

### 구현 정책
- `worker` 화면 상단은 카드와 배지 중심으로 간단하게 유지한다.
- 즉시 조치와 직접 연결되지 않는 복잡한 분석형 차트는 하단 보조 영역으로 둔다.
- 불량 원인 분석보다 “지금 멈춰야 하는가 / 점검해야 하는가”에 초점을 둔다.
- 최대 정보량은 상단 KPI/배지 4~5개, 로그 1개, 보조 차트 1개 수준으로 제한한다.
- 고도화 표현은 판단을 빠르게 돕는 범위 내에서만 허용한다.

---

## 4.2 품질 대시보드 (SCR-002 / qa)

### 화면 목적
품질 대시보드는 현재 품질 수준, 주요 불량 원인, 재검 대상을 함께 판단하는 화면이다.  
즉, 현재 상태, 원인 분석, 상세 판정 조회가 동시에 가능해야 한다.

### 1차 구현 기준
#### P1
- 현재 불량률: KPI 카드
- 주요 불량 유형 Top 5: 수평 바차트
- 시간대별 불량률 추이: 라인차트
- 재검 필요 수량: KPI 카드
- NG 검출 리스트: 상세 테이블
- 점선 예측 구간 시각화
- 이미지 썸네일 확장
- 판정 보조 인사이트 문구

#### P2
- 검사 현황 요약: 묶음형 정보 카드
- 신뢰도/판정 상태 배지 규칙 강화
- 전일 비교 또는 기준선 강조
- 예측 불량률 카드

#### P3
- 이미지 확장 UX 정교화
- 기준선/예측선 시각 강화
- 상세 인사이트 영역 확장

### 구현 정책
- `qa` 화면은 카드보다 차트와 테이블 비중을 높인다.
- 불량 유형, 추세, 판정 리스트의 연결성이 유지되도록 구성한다.
- 예측형 지표는 실제 측정값과 시각적으로 분리한다.
- 최대 정보량은 상단 KPI 2~3개, 차트 2개, 테이블 1개 수준으로 제한한다.
- 이미지, 예측, 인사이트는 실제 ML/LLM 연동이 아니라 1차에서는 simulation 또는 rule-based 방식으로 반영한다.

---

## 4.3 통합 대시보드 (SCR-003 / manager)

### 화면 목적
통합 대시보드는 전사 관점의 성과 비교와 운영 판단이 핵심이다.  
관리자는 전체 효율, 계획 대비 실적, 라인 간 편차, 운영 리스크를 빠르게 파악해야 한다.

### 1차 구현 기준
#### P1
- 전체 라인 OEE: KPI 카드
- 공장 전체 목표 달성률: KPI 카드 또는 프로그레스 카드
- 예상 종료 생산량: KPI 카드
- 라인별 생산량 비교: 바차트
- 시간대별 생산량 / 불량률 추이: 복수 시계열 라인차트
- 단순 게이지형 OEE 강조 표현
- 병목 흐름 표현용 단순 시각화
- 위험도/요약 인사이트 문구

#### P2
- 라인 상태 분포: 도넛차트
- 다운타임/손실 원인 비교: 바차트 또는 파레토 대체 구조
- 보조 KPI 1개 추가
- 불량 위험도 예측 카드
- 2D 디지털 트윈형 상태 보드

#### P3
- 예측 구간 정교화
- 흐름 시각화 확장
- 비교 시각화 고도화

### 구현 정책
- `manager` 화면은 상단 KPI 기본 3개, 최대 4개까지만 허용한다.
- 관리자는 한 라인보다 공장 전체 흐름과 비교를 봐야 하므로 단일 상세보다 집계와 비교를 우선한다.
- OEE와 목표 달성률은 전사 집계 기준으로 명확히 구분한다.
- 최대 정보량은 상단 KPI 3~4개, 비교 차트 2~3개 수준으로 제한한다.
- 병목 흐름, 위험도, 상태 보드는 1차에서 경량 버전으로 우선 구현한다.

---

## 4.4 생산 현황 보드 (SCR-004 / promo)

### 화면 목적
생산 현황 보드는 공용 송출 화면으로, 복잡한 분석보다 원거리 시인성과 즉시 이해 가능성이 우선이다.  
세부 해석보다 핵심 수치와 주의할 품질 이슈를 짧게 전달해야 한다.

### 1차 구현 기준
#### P1
- 누적 총 생산량: 대형 KPI 카드
- 목표 달성률: 대형 KPI 카드
- 당일 양품률: KPI 카드
- 시간대별 생산 추이: 단순 라인차트
- KPI 숫자 카운트업 애니메이션
- 자동 롤링 문구
- 자동 페이지 전환

#### P2
- TOP 3 Defect Class: 요약형 리스트 또는 단순 바차트
- 자동 시계
- 강조 색상 조정
- 대형 상태 메시지 블록

#### P3
- 애니메이션 강화
- 전광판형 연출 정교화
- 보조 장면 전환 개선

### 구현 정책
- `promo` 화면은 멀리서도 읽히는 숫자와 짧은 텍스트 중심으로 구성한다.
- 긴 테이블, 복잡한 범례, 작은 축 라벨은 사용하지 않는다.
- 결함 정보는 보조 수준으로만 노출하고, 생산/달성률/양품률을 우선한다.
- 최대 정보량은 대형 KPI 3개와 단순 추세 1개 수준으로 제한한다.
- 자동 롤링, 자동 전환, 카운트업은 시인성 강화 목적 범위 내에서 허용한다.

---

## 5. 데이터 표현 정책

## 5.1 데이터 유형 구분 목적
현재 프로젝트는 DB 구조를 즉시 확장하기 어려운 조건이 있으므로, 화면에 표현되는 지표를 생성 방식 기준으로 분리 관리해야 한다.  
이를 위해 모든 KPI, 차트, 인사이트는 아래 3개 `sourceType` 중 하나를 반드시 가진다.

- `actual`
- `derived`
- `simulated`

---

## 5.2 actual 데이터 정의

### 정의
DB 또는 API 응답에서 직접 조회되거나, 단순 집계만으로 바로 산출 가능한 값

### 포함 기준
- 원본 테이블 또는 API 응답에 직접 존재
- 해석상 추가 가정이 없음
- 같은 시점에 재조회하면 동일 결과를 기대할 수 있음

### 예시
- 최근 10분 NG 발생 건수
- 설비 가동 상태
- 누적 생산량
- 총 검사 수량
- 양품 수
- 불량 수
- 라인별 생산량 비교의 원본 값

### 정책
- `actual` 값은 화면의 기본 진실값으로 간주한다.
- 상태 배지, KPI 카드, 로그 리스트, 상세 테이블의 우선 입력값으로 사용한다.

---

## 5.3 derived 데이터 정의

### 정의
원본 데이터를 조합하거나 규칙 기반 계산을 적용해 산출한 파생 지표

### 포함 기준
- `actual` 데이터 조합으로 계산 가능
- 계산식과 기준 시점이 명확함
- 추가 추론 없이 재현 가능함

### 예시
- 현재 불량률
- 목표 달성률
- OEE
- 시간대별 불량률
- 라인 상태 분포
- Top N Defect 집계
- 예상 종료 생산량(현재 속도 기준 단순 추정)

### 정책
- `derived` 값은 계산식과 기준 필드를 문서화해야 한다.
- `actual`과 혼동되지 않도록 계산값 성격을 내부적으로 구분한다.
- 계산식이 불안정한 경우 카드보다 보조 지표로 우선 배치한다.
- 계산 책임은 가능하면 BE/API 우선으로 하고, 불가 시 FE 계산으로 처리한다.

---

## 5.4 simulated 데이터 정의

### 정의
현재 DB/API 범위로는 직접 확보되지 않지만, 화면 완성도와 2차 구현 표현을 위해 규칙 기반 또는 시범형 형태로 생성한 값

### 포함 기준
- 실제 저장 데이터가 아님
- 화면 시연 또는 고도화 개념 표현 목적
- 가정, 버전, 유효기간, 주의 문구가 함께 제공됨

### 예시
- 예측 불량률
- 불량 위험도 예측
- 재산출 필요 표시가 있는 인사이트 카드
- 룰 기반 추천 또는 리스크 경고 문구
- mock inference 기반 요약 텍스트
- 2D 디지털 트윈형 상태 표현

### 정책
- `simulated` 값은 반드시 `sourceType: simulated`를 유지한다.
- 실제 운영 KPI와 동일 위치 및 강도로 표시하지 않는다.
- 아래 메타를 반드시 포함한다.
  - `sourceType`
  - `assumption`
  - `validUntil`
  - `modelVersion`
  - `caution`
- 카드 제목, 보조 라벨, 배지 중 최소 2개 이상으로 실제값과 시각적으로 분리한다.
- 유효기간 만료 시 재산출 필요 상태로 전환한다.
- 1차 구현에서는 실제 모델 서빙이 아니라 `simulation layer`, `rule-based logic`, `mock inference` 방식으로 구현할 수 있다.

---

## 6. 상태 처리 및 단위 정책

## 6.1 데이터 상태 처리 규칙
모든 공통 컴포넌트는 아래 상태를 기본 처리해야 한다.

- `null`: `N/A` 또는 `데이터 없음`
- `0`: 정상 값으로 표시
- `empty array`: 빈 상태 UI 표시
- `error`: 오류 문구 또는 fallback block 표시
- `stale`: `최신 아님` 배지 또는 갱신 시각 표시

상태 처리는 공통 컴포넌트에서 1차 처리하고, 화면별 호출부는 문구만 override한다.

## 6.2 단위 표기 규칙
문서와 UI 전체에서 아래 기준을 통일한다.

- 비율: `%`
- 개수: `건`
- 수량: `ea` 또는 `pcs` 중 1종으로 통일
- 시간: `분`

## 6.3 시간 기준 규칙
- 최근 10분: 현재 시각 기준 직전 10분
- 당일: 로컬 날짜 00:00~현재
- 시간대별: 당일 시간 버킷 기준
- 전일 비교: 전일 동일 시간대 기준

---

## 7. 컴포넌트별 적용 정책

## 7.1 KPI 카드

### 적용 대상
- 단일 수치의 현재 상태 전달
- 즉시 해석이 필요한 핵심 지표

### 우선 적용 화면
- worker
- qa
- manager
- promo

### 사용 원칙
- 상단 요약 영역에 배치한다.
- 상태 텍스트, 보조 문구, 단위 표기를 일관되게 적용한다.
- 화면별 최대 카드 수를 초과하지 않는다.
- `simulated` 카드에는 별도 식별 표기를 반드시 포함한다.
- 필요 시 경량 스파크라인, 카운트업, 단순 게이지 표현을 보조로 포함할 수 있다.

---

## 7.2 상태 배지

### 적용 대상
- 설비 상태
- 판정 상태
- 로그 상태
- 위험도 단계

### 사용 원칙
- `RUN / IDLE / DOWN`, `PASS / NG / CHECK` 등 상태 체계를 표준화한다.
- 색상은 상태 의미를 전달하는 범위에서만 제한적으로 사용한다.
- 본문 강조용 장식 색상으로 남용하지 않는다.

---

## 7.3 프로그레스 바 및 게이지형 표현

### 적용 대상
- 목표 달성률
- 잔량/진척도
- 기준 대비 현황
- OEE 강조 표현

### 사용 원칙
- `worker`, `manager`, `promo` 화면에서 우선 사용한다.
- 1차 구현에서는 복합 게이지가 아니라 단순 진행률형, 반원형, 도넛형 수준으로 제한한다.
- 색상보다 진행률 해석을 우선한다.

---

## 7.4 라인차트

### 적용 대상
- 시간대별 생산량 추이
- 시간대별 불량률 추이
- 목표선 또는 전일 비교
- 예측 구간 표시

### 사용 원칙
- 시계열 해석이 필요한 영역에만 사용한다.
- `promo` 화면은 축과 범례를 단순화한다.
- 예측값은 실측값과 선 스타일 또는 영역을 분리한다.
- 축 범위 이상치, 데이터 없음, 단위 불일치 처리를 공통 옵션에서 관리한다.
- 점선 예측 구간은 simulated 데이터에만 적용한다.

---

## 7.5 바차트

### 적용 대상
- 주요 불량 유형 Top N
- 라인별 생산량 비교
- 원인별 비교
- 단계별 병목 비교

### 사용 원칙
- 긴 라벨은 수평 바차트 우선으로 한다.
- 순위 비교는 bar가 donut보다 우선이다.
- 불량 유형 Top 3/5 구조에 재사용 가능해야 한다.
- 항목 수가 과도하면 Top N으로 절단하고 나머지는 기타로 처리할 수 있다.

---

## 7.6 도넛차트

### 적용 대상
- 구성비 또는 상태 분포
- 라인 상태 분포
- 검사 결과 비중

### 사용 원칙
- 비율 전달이 목적일 때만 사용한다.
- 항목 수가 많으면 bar로 대체한다.
- 표시는 단순하게 유지하고 범례는 보조 정보로 사용한다.

---

## 7.7 로그 리스트

### 적용 대상
- 실시간 NG 로그
- 최근 이벤트
- 알람 이력

### 사용 원칙
- `worker` 화면 우선으로 적용한다.
- 최신순 정렬을 유지한다.
- 시간 / 라인 / 상태 / 메시지 중심의 최소 정보만 표시한다.
- 조치 판단이 가능할 정도의 정보만 남기고 과도한 상세값은 제외한다.
- promo에서는 세부 로그 대신 롤링 요약 문구로 축약할 수 있다.

---

## 7.8 상세 테이블 및 썸네일 확장

- 상세 품질 이력
- 이미지 기반 검출 결과

### 사용 원칙

- `datetime / id / status / percent` 유형별 셀 렌더링 규칙을 통일한다.
- 긴 텍스트는 `title` 보조 처리, 숫자는 우측 정렬한다.
- 이미지 썸네일은 1차에서 모달 확장 수준으로 구현한다.
- 상세 테이블은 신규 발명보다 기존 구조 정리를 우선한다.

## 7.9 인사이트 문구 및 자동 전환 표현

- 예측 요약

### 사용 원칙

- actual 지표를 덮어쓰지 않고 보조 해석 영역으로만 사용한다.
- promo 화면의 자동 롤링 및 자동 페이지 전환은 시인성 강화 목적 범위로 제한한다.

## 9. 고도화 요소 반영 기준 및 제외 범위

### 9.1 1차 구현에 포함하는 고도화 표현 항목
다음 항목은 화면 완성도와 2차 구현 수준의 인상을 확보하기 위해 1차 구현 범위에 포함한다.  
단, 실제 고도화 엔진 전체를 구현하는 것이 아니라 경량 표현 또는 시뮬레이션 방식으로 우선 반영한다.

- 게이지 차트
- 스파크라인 카드
- 카드 카운트업 애니메이션
- 자동 롤링 문구
- 자동 페이지 전환
- 이미지 썸네일 확장
- 병목 흐름 표현용 단순 시각화
- 점선 예측 구간 시각화
- 예측/위험도/요약용 simulated 인사이트 문구
- 디지털 트윈형 2D 상태 표현

### 9.2 1차 구현에서의 반영 방식
- 게이지 차트는 복합형이 아니라 단순 진행률형으로 구현한다.
- 스파크라인 카드는 KPI 보조 추세선 수준으로 제한한다.
- 카운트업 애니메이션은 핵심 KPI에만 제한 적용한다.
- 자동 롤링과 자동 페이지 전환은 promo 화면 중심으로 우선 적용한다.
- 이미지 썸네일은 qa 화면에서 모달 확장 수준으로 구현한다.
- 병목 흐름 시각화는 복잡한 샌키 대신 단순 단계 비교형으로 우선 구현한다.
- 점선 예측 구간은 simulated 데이터에만 적용한다.
- 인사이트 문구는 실제 생성형 AI 연동이 아니라 룰 기반 문장 생성 또는 mock inference 방식으로 구현한다.
- 디지털 트윈형 표현은 3D 구현이 아니라 라인/설비 상태를 보여주는 2D 보드 수준으로 구현한다.

### 9.3 1차 구현에서 제외하는 항목
다음 항목은 1차 범위에서 제외한다.

- 실제 모델 학습/서빙 파이프라인 구축
- 외부 생성형 AI 연동
- 3D 디지털 트윈 엔진
- 고복잡도 멀티축 분석 대시보드 전면 적용
- 모든 차트를 하나의 함수로 강제 통합하는 수준의 과도한 공통화

### A-1. 역할별 정보 위계
- 작업자 화면은 즉시 조치 중심
- 품질 화면은 원인 분석과 판정 중심
- 관리자 화면은 비교와 자원 배분 중심
- 공용 보드는 시인성과 공유 중심

### A-2. ISA-101 / 고성능 HMI 반영 원칙
- 정상 상태는 절제되게 표현
- 이상 상태는 빠르게 식별되도록 강조
- 과도한 장식, 불필요한 애니메이션, 복잡한 3D 연출은 지양
- 숫자는 맥락과 함께 보여주고, 해석이 빠른 형태를 우선 적용

### A-3. 본 프로젝트 적용 결론
- 1차 구현은 카드, 배지, 프로그레스, 라인차트, 바차트, 로그, 테이블 중심으로 간다.
- 동시에 화면 완성도 향상에 필요한 경량 고도화 표현을 제한적으로 포함한다.
- 예측 및 인사이트성 요소는 `simulated` 계층으로 분리한다.
- 현재 DB/API 범위를 벗어나는 실제 엔진 구현은 보류하되, 화면 구조상 자리는 남겨둘 수 있다.
- 공통화는 재사용성과 유지보수성을 높이는 범위까지만 적용하고, 화면 역할 차이는 보존한다.

## 12. 최종 결론

본 단계의 시각화 구현은 많이 보여주는 것보다 역할에 맞는 핵심 정보를 빠르게 해석하게 하는 것을 목표로 한다.  
따라서 1차 구현은 `actual / derived` 중심의 표준 시각화를 기반으로 완성도를 확보하고, `simulated` 지표와 경량 고도화 표현을 활용해 2차 구현 수준의 인상을 부분적으로 반영한다.

이는 실제 예측 엔진, 디지털 트윈 엔진, 외부 생성형 AI 연동을 완성하는 것을 의미하지 않는다.  
1차 단계에서는 `simulation layer`, `rule-based calculation`, `mock inference`, 경량 인터랙션을 활용한 표현 중심 구현으로 한정한다.

이 기준에 따라 후속 구현은 공통 토큰과 공통 규칙을 먼저 정리한 뒤, `worker → qa → manager → promo` 순으로 화면별 바인딩을 진행한다.  
코드 생성 단계에서는 `P1` 항목을 우선 구현하고, `P2`와 `P3`는 범위 관리 하에 순차적으로 확장한다.

# 9.5.3.11 더미데이터 바인딩 적용 정리

## 1. 작업 목적
- 1차 구현 범위 기준으로 대시보드 더미데이터 바인딩을 정리하고 화면별 렌더/예외 처리 기준을 고정한다.

## 2. 기준 문서
- 기준 고정: `9.5.3.1_차트테이블_구현_대상_목록.md`
- 보조 참조: `9.5.3.2` ~ `9.5.3.10`

## 3. 1차 구현 확정 범위
- SCR-001: 최근 10분 NG 발생 건수, 목표 달성률, 누적 생산량, 설비 가동 상태, 실시간 NG 로그, 최근 발생 유형 Top 3
- SCR-002: 현재 불량률, 검사 현황 요약, 주요 불량 유형 Top 5, 시간대별 불량률 추이, NG 검출 리스트
- SCR-003: 공장 목표 달성률, 라인별 생산량 비교, 시간대별 생산·불량 추이
- SCR-004: 누적 총 생산량, 당일 양품률, 목표 달성률, TOP 3 Defect Class, 시간대별 생산 추이

## 4. 제외 범위(이번 단계 미포함)
- Worker 도넛차트
- Worker 라인차트
- Manager 도넛차트
- Manager 상세 테이블
- QA 도넛차트
- Manager 게이지/위험도 예측
- QA 예측 불량률
- Manager 예상 종료량
- OEE
- 재검 필요 수량
- 설비 상태의 equipment.status 기반 해석
- PASS/CHECK 포함 전체 검사 결과 테이블

## 5. 적용 원칙
- 렌더 흐름은 `fetchAll -> activateTab -> render* / render*Chart` 유지
- KPI 스키마는 화면별 `key/title/value/unit/helperText/trendType/trendValue/status/statusText`로 통일
- 상태 스키마는 `statusData.worker.equipment_status` 구조로 통일
- 단위는 문자열 결합이 아닌 `value + unit` 분리 원칙 적용
- 예외 처리 기준:
- `0`은 정상 출력
- `null/undefined`는 fallback(`-` 또는 `데이터 없음`) 출력
- `[]`는 empty-state 문구 출력

## 6. 실제 변경 파일
- `nginx/html/js/wed_dashboard/dashboard_api.js`
- `nginx/html/js/wed_dashboard/dashboard_charts.js`
- `nginx/html/js/wed_dashboard/dashboard_app.js`
- `nginx/html/web_dashboard.html`
- `nginx/html/css/wed_dashboard/dashboard_style.css`

## 7. 반영 핵심
- API:
- `fetchAll` 반환 키를 `summary/lines/defects/hourly/ngList/kpiCards/statusData`로 정리
- `MOCK_KPI_CARDS`를 화면별 객체(`worker/qa/manager/promo`)로 구성
- `MOCK_NG_LIST`는 기존 키(`time/board_id/cls/conf`) 유지
- Charts:
- `renderCanvasEmpty` 추가
- `renderWorkerTopDefectBarChart` 추가
- QA/Manager/Promo 차트 함수에 empty-state 및 `0`값 처리 반영
